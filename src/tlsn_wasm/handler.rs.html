1//! WASM bindings for handler-based range extraction.
2//!
3//! Exposes [`compute_reveal`] to JavaScript, which parses HTTP transcripts
4//! and maps plugin handlers to byte ranges for selective disclosure.
5
6use wasm_bindgen::prelude::*;
7
8/// Parses HTTP request/response transcripts and maps handlers to byte ranges.
9///
10/// This is the WASM wrapper around `tlsn_sdk_core::compute_reveal`.
11///
12/// # Arguments
13///
14/// * `sent` - Raw bytes of the HTTP request (sent data).
15/// * `recv` - Raw bytes of the HTTP response (received data).
16/// * `handlers` - Array of handler objects (deserialized from JS).
17///
18/// # Returns
19///
20/// A `ComputeRevealOutput` object containing:
21/// - `sentRanges` / `recvRanges`: byte ranges for `Prover.reveal()`
22/// - `sentRangesWithHandlers` / `recvRangesWithHandlers`: ranges annotated with
23///   handlers
24#[wasm_bindgen(js_name = compute_reveal)]
25pub fn compute_reveal(sent: &[u8], recv: &[u8], handlers: JsValue) -> Result<JsValue, JsError> {
26    let handlers: Vec<tlsn_sdk_core::Handler> = serde_wasm_bindgen::from_value(handlers)
27        .map_err(|e| JsError::new(&format!("failed to deserialize handlers: {e}")))?;
28
29    let output = tlsn_sdk_core::compute_reveal(sent, recv, &handlers)
30        .map_err(|e| JsError::new(&e.to_string()))?;
31
32    serde_wasm_bindgen::to_value(&output).map_err(|e| JsError::new(&e.to_string()))
33}
