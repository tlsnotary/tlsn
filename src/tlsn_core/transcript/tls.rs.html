95impl TlsTranscript {
96    /// Creates a new TLS transcript.
97    #[allow(clippy::too_many_arguments)]
98    pub fn new(
99        time: u64,
100        version: TlsVersion,
101        server_cert_chain: Option<Vec<CertificateDer>>,
102        server_signature: Option<ServerSignature>,
103        certificate_binding: CertBinding,
104        plain_verify_data: Option<VerifyData>,
105        sent: Vec<Record>,
106        recv: Vec<Record>,
107        cf_vd: Record,
108        sf_vd: Record,
109    ) -> Result<Self, TlsTranscriptError> {
110        // Check for client finished consistency if possible.
111        if let Some(verify_data) = &plain_verify_data {
112            let payload = cf_vd
113                .plaintext
114                .as_ref()
115                .ok_or(TlsTranscriptError::validation(
116                    "client finished message was hidden from the follower",
117                ))?;
118
119            let mut reader = Reader::init(payload);
120            let payload =
121                HandshakeMessagePayload::read_version(&mut reader, ProtocolVersion::TLSv1_2)
122                    .ok_or(TlsTranscriptError::validation(
123                        "first record sent was not a handshake message",
124                    ))?;
125
126            let HandshakePayload::Finished(vd) = payload.payload else {
127                return Err(TlsTranscriptError::validation(
128                    "first record sent was not a client finished message",
129                ));
130            };
131
132            if vd.0 != verify_data.client_finished {
133                return Err(TlsTranscriptError::validation(
134                    "inconsistent client finished verify data",
135                ));
136            }
137        }
138
139        // Check for server_finished finished consistency if possible.
140        if let Some(verify_data) = &plain_verify_data {
141            let payload = sf_vd
142                .plaintext
143                .as_ref()
144                .ok_or(TlsTranscriptError::validation(
145                    "server finished message was hidden from the follower",
146                ))?;
147
148            let mut reader = Reader::init(payload);
149            let payload =
150                HandshakeMessagePayload::read_version(&mut reader, ProtocolVersion::TLSv1_2)
151                    .ok_or(TlsTranscriptError::validation(
152                        "first record received was not a handshake message",
153                    ))?;
154
155            let HandshakePayload::Finished(vd) = payload.payload else {
156                return Err(TlsTranscriptError::validation(
157                    "first record received was not a server finished message",
158                ));
159            };
160
161            if vd.0 != verify_data.server_finished {
162                return Err(TlsTranscriptError::validation(
163                    "inconsistent server finished verify data",
164                ));
165            }
166        }
167
168        let mut sent_iter = sent.iter();
169        let mut recv_iter = recv.iter();
170
171        // Verify last record sent was either application data or close notify.
172        if let Some(record) = sent_iter.next_back() {
173            match record.typ {
174                ContentType::ApplicationData => {}
175                ContentType::Alert => {
176                    // Ensure the alert is a close notify.
177                    let payload =
178                        record
179                            .plaintext
180                            .as_ref()
181                            .ok_or(TlsTranscriptError::validation(
182                                "alert content was hidden from the follower",
183                            ))?;
184
185                    let mut reader = Reader::init(payload);
186                    let payload = AlertMessagePayload::read(&mut reader).ok_or(
187                        TlsTranscriptError::validation("alert message was malformed"),
188                    )?;
189
190                    let AlertDescription::CloseNotify = payload.description else {
191                        return Err(TlsTranscriptError::validation(
192                            "sent alert that is not close notify",
193                        ));
194                    };
195                }
196                typ => {
197                    return Err(TlsTranscriptError::validation(format!(
198                        "sent unexpected record content type: {typ:?}"
199                    )));
200                }
201            }
202        }
203
204        // Verify last record received was either application data or close notify.
205        if let Some(record) = recv_iter.next_back() {
206            match record.typ {
207                ContentType::ApplicationData => {}
208                ContentType::Alert => {
209                    // Ensure the alert is a close notify.
210                    let payload =
211                        record
212                            .plaintext
213                            .as_ref()
214                            .ok_or(TlsTranscriptError::validation(
215                                "alert content was hidden from the follower",
216                            ))?;
217
218                    let mut reader = Reader::init(payload);
219                    let payload = AlertMessagePayload::read(&mut reader).ok_or(
220                        TlsTranscriptError::validation("alert message was malformed"),
221                    )?;
222
223                    let AlertDescription::CloseNotify = payload.description else {
224                        return Err(TlsTranscriptError::validation(
225                            "received alert that is not close notify",
226                        ));
227                    };
228                }
229                typ => {
230                    return Err(TlsTranscriptError::validation(format!(
231                        "received unexpected record content type: {typ:?}"
232                    )));
233                }
234            }
235        }
236
237        // Ensure all other records were application data.
238        for record in sent_iter {
239            if record.typ != ContentType::ApplicationData {
240                return Err(TlsTranscriptError::validation(format!(
241                    "sent unexpected record content type: {:?}",
242                    record.typ
243                )));
244            }
245        }
246
247        for record in recv_iter {
248            if record.typ != ContentType::ApplicationData {
249                return Err(TlsTranscriptError::validation(format!(
250                    "received unexpected record content type: {:?}",
251                    record.typ
252                )));
253            }
254        }
255
256        Ok(Self {
257            time,
258            version,
259            server_cert_chain,
260            server_signature,
261            certificate_binding,
262            sent,
263            recv,
264            cf_vd,
265            sf_vd,
266            cf_hash: None,
267            session_hash: None,
268            sf_hash_input: None,
269        })
270    }

337    /// Returns the server finished hash.
338    pub fn sf_hash(&self, cf_vd: &[u8; 12]) -> Option<Vec<u8>> {
339        let SfHashInput {
340            sent_hs_bytes,
341            recv_hs_bytes,
342            sent_ch_end,
343            recv_shd_end,
344        } = self.sf_hash_input.as_ref()?;
345
346        let mut hasher = Sha256::new();
347        hasher.update(&sent_hs_bytes[..*sent_ch_end]);
348        hasher.update(&recv_hs_bytes[..*recv_shd_end]);
349        hasher.update(&sent_hs_bytes[*sent_ch_end..]);
350        // Append the reconstructed Client Finished handshake message.
351        hasher.update([0x14, 0x00, 0x00, 0x0c]);
352        hasher.update(cf_vd);
353        hasher.update(&recv_hs_bytes[*recv_shd_end..]);
354
355        Some(hasher.finalize().to_vec())
356    }

358    /// Parses a complete TLS transcript from raw wire bytes.
359    ///
360    /// # Arguments
361    ///
362    /// * `time` - UNIX timestamp of the connection.
363    /// * `sent` - Raw TLS bytes sent (client → server).
364    /// * `recv` - Raw TLS bytes received (server → client).
365    /// * `sent_app` - Plaintext application data sent.
366    /// * `recv_app` - Plaintext application data received.
367    pub fn parse(
368        time: u64,
369        sent: &[u8],
370        recv: &[u8],
371        sent_app: &[u8],
372        recv_app: &[u8],
373    ) -> Result<Self, TlsTranscriptError> {
374        let sent_records = parse_raw_records(sent)?;
375        let recv_records = parse_raw_records(recv)?;
376
377        let (sent_hs_bytes, sent_hs) = parse_handshake_stream(&sent_records)?;
378        let (recv_hs_bytes, recv_hs) = parse_handshake_stream(&recv_records)?;
379
380        let (version, handshake) = extract_handshake(&sent_hs, &recv_hs)?;
381
382        let sent_msgs = scan_handshake_messages(&sent_hs_bytes)?;
383        let recv_msgs = scan_handshake_messages(&recv_hs_bytes)?;
384        let (cf_hash, session_hash, sf_hash_input) =
385            compute_handshake_hashes(sent_hs_bytes, recv_hs_bytes, &sent_msgs, &recv_msgs)?;
386
387        let sent_vd = parse_verify_data_record(&sent_records)?;
388        let recv_vd = parse_verify_data_record(&recv_records)?;
389
390        let sent_app_records = parse_app_records(&sent_records, sent_app)?;
391        let recv_app_records = parse_app_records(&recv_records, recv_app)?;
392
393        let mut transcript = Self::new(
394            time,
395            version,
396            Some(handshake.certs),
397            Some(handshake.sig),
398            handshake.binding,
399            None,
400            sent_app_records,
401            recv_app_records,
402            sent_vd,
403            recv_vd,
404        )?;
405        transcript.cf_hash = Some(cf_hash.to_vec());
406        transcript.session_hash = Some(session_hash.to_vec());
407        transcript.sf_hash_input = Some(sf_hash_input);
408
409        Ok(transcript)
410    }

412    /// Returns the application data transcript.
413    pub fn to_transcript(&self) -> Result<Transcript, TlsTranscriptError> {
414        let mut sent = Vec::new();
415        let mut recv = Vec::new();
416
417        for record in self
418            .sent
419            .iter()
420            .filter(|record| record.typ == ContentType::ApplicationData)
421        {
422            let plaintext = record
423                .plaintext
424                .as_ref()
425                .ok_or(ErrorRepr::Incomplete {
426                    direction: Direction::Sent,
427                    seq: record.seq,
428                })?
429                .clone();
430            sent.extend_from_slice(&plaintext);
431        }
432
433        for record in self
434            .recv
435            .iter()
436            .filter(|record| record.typ == ContentType::ApplicationData)
437        {
438            let plaintext = record
439                .plaintext
440                .as_ref()
441                .ok_or(ErrorRepr::Incomplete {
442                    direction: Direction::Received,
443                    seq: record.seq,
444                })?
445                .clone();
446            recv.extend_from_slice(&plaintext);
447        }
448
449        Ok(Transcript::new(sent, recv))
450    }

504/// Parse raw TLS record frames from a byte slice.
505fn parse_raw_records(bytes: &[u8]) -> Result<Vec<OpaqueMessage>, TlsTranscriptError> {
506    let mut reader = Reader::init(bytes);
507    let mut records = Vec::new();
508    while reader.any_left() {
509        let msg = OpaqueMessage::read(&mut reader)
510            .map_err(|e| TlsTranscriptError::parse(format!("failed to read TLS record: {e:?}")))?;
511        records.push(msg);
512    }
513    Ok(records)
514}

516/// Collect the pre-CCS handshake byte stream and decode it into
517/// individual handshake messages.
518fn parse_handshake_stream(
519    records: &[OpaqueMessage],
520) -> Result<(Vec<u8>, Vec<HandshakeMessagePayload>), TlsTranscriptError> {
521    let handshake_bytes = collect_handshake_bytes_pre_ccs(records);
522
523    let mut reader = Reader::init(&handshake_bytes);
524    let mut messages = Vec::new();
525    while reader.any_left() {
526        let msg = HandshakeMessagePayload::read_version(&mut reader, ProtocolVersion::TLSv1_2)
527            .ok_or_else(|| TlsTranscriptError::parse("failed to parse handshake message"))?;
528        messages.push(msg);
529    }
530    Ok((handshake_bytes, messages))
531}

533/// Validate handshake-message bounds and compute the two
534/// digests `cf_hash` and `session_hash`, and the cached
535/// inputs needed to derive `sf_hash` later from `cf_vd`.
536fn compute_handshake_hashes(
537    sent_hs_bytes: Vec<u8>,
538    recv_hs_bytes: Vec<u8>,
539    sent_msgs: &[HandshakeMsgInfo],
540    recv_msgs: &[HandshakeMsgInfo],
541) -> Result<([u8; 32], [u8; 32], SfHashInput), TlsTranscriptError> {
542    let client_hello = sent_msgs
543        .first()
544        .ok_or_else(|| TlsTranscriptError::parse("missing ClientHello in sent handshake stream"))?;
545    if client_hello.typ != HandshakeType::ClientHello {
546        return Err(TlsTranscriptError::parse(
547            "first sent handshake message is not ClientHello",
548        ));
549    }
550    let ckx = sent_msgs
551        .iter()
552        .find(|m| m.typ == HandshakeType::ClientKeyExchange)
553        .ok_or_else(|| TlsTranscriptError::parse("missing ClientKeyExchange"))?;
554    if !recv_msgs
555        .iter()
556        .any(|m| m.typ == HandshakeType::ServerHello)
557    {
558        return Err(TlsTranscriptError::parse("missing ServerHello"));
559    }
560    let shd = recv_msgs
561        .iter()
562        .find(|m| m.typ == HandshakeType::ServerHelloDone)
563        .ok_or_else(|| TlsTranscriptError::parse("missing ServerHelloDone"))?;
564
565    let sent_ch_end = client_hello.end;
566    let recv_shd_end = shd.end;
567
568    // session_hash: ClientHello → server flight (ServerHello..ServerHelloDone)
569    // → client flight up to and including ClientKeyExchange.
570    let mut hasher = Sha256::new();
571    hasher.update(&sent_hs_bytes[..sent_ch_end]);
572    hasher.update(&recv_hs_bytes);
573    hasher.update(&sent_hs_bytes[sent_ch_end..ckx.end]);
574    let session_hash: [u8; 32] = hasher.finalize().into();
575
576    // cf_hash: ClientHello → server first flight (..ServerHelloDone)
577    // → remaining client flight (ClientKeyExchange and, when client
578    // auth is active, CertificateVerify).
579    let mut hasher = Sha256::new();
580    hasher.update(&sent_hs_bytes[..sent_ch_end]);
581    hasher.update(&recv_hs_bytes[..recv_shd_end]);
582    hasher.update(&sent_hs_bytes[sent_ch_end..]);
583    let cf_hash: [u8; 32] = hasher.finalize().into();
584
585    Ok((
586        cf_hash,
587        session_hash,
588        SfHashInput {
589            sent_hs_bytes,
590            recv_hs_bytes,
591            sent_ch_end,
592            recv_shd_end,
593        },
594    ))
595}

597/// Concatenate the payload bytes of all handshake records appearing
598/// before the first ChangeCipherSpec. These are the plaintext
599/// handshake bytes as they appeared on the wire.
600fn collect_handshake_bytes_pre_ccs(records: &[OpaqueMessage]) -> Vec<u8> {
601    let mut handshake_bytes = Vec::new();
602    for record in records {
603        if record.typ == TlsContentType::ChangeCipherSpec {
604            break;
605        }
606        if record.typ == TlsContentType::Handshake {
607            handshake_bytes.extend_from_slice(&record.payload.0);
608        }
609    }
610    handshake_bytes
611}

613/// Position information for a single handshake message inside a
614/// concatenated handshake byte stream.
615struct HandshakeMsgInfo {
616    typ: HandshakeType,
617    /// Exclusive end offset of the message (header + body) in the
618    /// scanned byte stream.
619    end: usize,
620}
621
622/// Walk a concatenated handshake byte stream and return the
623/// [`HandshakeMsgInfo`].
624fn scan_handshake_messages(bytes: &[u8]) -> Result<Vec<HandshakeMsgInfo>, TlsTranscriptError> {
625    let mut out = Vec::new();
626    let mut pos = 0;
627    while pos < bytes.len() {
628        if bytes.len() - pos < 4 {
629            return Err(TlsTranscriptError::parse("truncated handshake header"));
630        }
631        let typ = HandshakeType::from(bytes[pos]);
632        let len = u32::from_be_bytes([0, bytes[pos + 1], bytes[pos + 2], bytes[pos + 3]]) as usize;
633        let msg_end = pos + 4 + len;
634        if msg_end > bytes.len() {
635            return Err(TlsTranscriptError::parse(
636                "handshake message length overflows buffer",
637            ));
638        }
639        out.push(HandshakeMsgInfo { typ, end: msg_end });
640        pos = msg_end;
641    }
642    Ok(out)
643}

645/// Find the index of the first ChangeCipherSpec record.
646fn find_ccs(records: &[OpaqueMessage]) -> Option<usize> {
647    records

649        .position(|r| r.typ == TlsContentType::ChangeCipherSpec)

652/// Split an encrypted TLS record payload into nonce / ciphertext / tag.
653fn split_into_record(
654    seq: u64,
655    payload: &[u8],
656    typ: TlsContentType,
657) -> Result<Record, TlsTranscriptError> {
658    let typ = ContentType::from(typ);
659
660    if payload.len() < NONCE_LEN + TAG_LEN {
661        return Err(TlsTranscriptError::parse("encrypted record too short"));
662    }
663
664    Ok(Record {
665        seq,
666        typ,
667        plaintext: None,
668        explicit_nonce: payload[..NONCE_LEN].to_vec(),
669        ciphertext: payload[NONCE_LEN..payload.len() - TAG_LEN].to_vec(),
670        tag: Some(payload[payload.len() - TAG_LEN..].to_vec()),
671    })
672}

674/// Extract the full handshake data from parsed handshake messages.
675fn extract_handshake(
676    sent_hs: &[HandshakeMessagePayload],
677    recv_hs: &[HandshakeMessagePayload],
678) -> Result<(TlsVersion, HandshakeData), TlsTranscriptError> {
679    let (version, server_random) = extract_server_hello_data(recv_hs)?;
680    let client_random = extract_client_random(sent_hs)?;
681    let certs = extract_certs(recv_hs)?;
682    let (server_ephemeral_key, sig) = extract_server_key_exchange(recv_hs, &certs)?;
683
684    let binding = CertBinding::V1_2(CertBindingV1_2 {
685        client_random,
686        server_random,
687        server_ephemeral_key,
688    });
689
690    let handshake = HandshakeData {
691        certs,
692        sig,
693        binding,
694    };
695
696    Ok((version, handshake))
697}

699/// Extract the TLS version and server random from the ServerHello message.
700fn extract_server_hello_data(
701    recv_hs: &[HandshakeMessagePayload],
702) -> Result<(TlsVersion, [u8; 32]), TlsTranscriptError> {
703    let server_hello = recv_hs
704        .iter()
705        .find_map(|msg| match &msg.payload {
706            HandshakePayload::ServerHello(sh) => Some(sh),
707            _ => None,
708        })
709        .ok_or_else(|| TlsTranscriptError::parse("missing ServerHello"))?;
710
711    let version = TlsVersion::try_from(server_hello.legacy_version)
712        .map_err(|e| TlsTranscriptError::parse(format!("unsupported TLS version: {e}")))?;
713
714    Ok((version, server_hello.random.0))
715}

723            HandshakePayload::ClientHello(ch) => Some(ch),

731fn extract_certs(
732    recv_hs: &[HandshakeMessagePayload],
733) -> Result<Vec<CertificateDer>, TlsTranscriptError> {
734    let cert_payload = recv_hs
735        .iter()
736        .find_map(|msg| match &msg.payload {
737            HandshakePayload::Certificate(certs) => Some(certs),
738            _ => None,
739        })
740        .ok_or_else(|| TlsTranscriptError::parse("missing Certificate"))?;
741
742    Ok(cert_payload
743        .iter()
744        .map(|cert| CertificateDer(cert.0.clone()))
745        .collect())
746}

748fn extract_server_key_exchange(
749    recv_hs: &[HandshakeMessagePayload],
750    certs: &[CertificateDer],
751) -> Result<(ServerEphemKey, ServerSignature), TlsTranscriptError> {
752    let ske = recv_hs
753        .iter()
754        .find_map(|msg| match &msg.payload {
755            HandshakePayload::ServerKeyExchange(ske) => Some(ske),
756            _ => None,
757        })
758        .ok_or_else(|| TlsTranscriptError::parse("missing ServerKeyExchange"))?;
759
760    let ecdhe = ske
761        .unwrap_given_kxa(&KeyExchangeAlgorithm::ECDHE)
762        .ok_or_else(|| TlsTranscriptError::parse("failed to parse ECDHE ServerKeyExchange"))?;
763
764    if ecdhe.params.curve_params.named_group != NamedGroup::secp256r1 {
765        return Err(TlsTranscriptError::parse(
766            "unsupported key exchange group (only secp256r1 is supported)",
767        ));
768    }
769
770    let key = ServerEphemKey {
771        typ: KeyType::SECP256R1,
772        key: ecdhe.params.public.0.clone(),
773    };
774
775    let alg = map_signature_scheme(ecdhe.dss.scheme, certs)?;
776    let sig = ServerSignature {
777        alg,
778        sig: ecdhe.dss.sig.0.clone(),
779    };
780
781    Ok((key, sig))
782}

784/// Map a TLS `SignatureScheme` to our `SignatureAlgorithm`.
785///
786/// For ECDSA in TLS 1.2 the scheme only specifies the hash, not the curve.
787/// The curve is determined from the end-entity certificate's public key.
788fn map_signature_scheme(
789    scheme: SignatureScheme,
790    certs: &[CertificateDer],
791) -> Result<SignatureAlgorithm, TlsTranscriptError> {
792    match scheme {
793        SignatureScheme::RSA_PKCS1_SHA256 => Ok(SignatureAlgorithm::RSA_PKCS1_2048_8192_SHA256),
794        SignatureScheme::RSA_PKCS1_SHA384 => Ok(SignatureAlgorithm::RSA_PKCS1_2048_8192_SHA384),
795        SignatureScheme::RSA_PKCS1_SHA512 => Ok(SignatureAlgorithm::RSA_PKCS1_2048_8192_SHA512),
796        SignatureScheme::RSA_PSS_SHA256 => {
797            Ok(SignatureAlgorithm::RSA_PSS_2048_8192_SHA256_LEGACY_KEY)
798        }
799        SignatureScheme::RSA_PSS_SHA384 => {
800            Ok(SignatureAlgorithm::RSA_PSS_2048_8192_SHA384_LEGACY_KEY)
801        }
802        SignatureScheme::RSA_PSS_SHA512 => {
803            Ok(SignatureAlgorithm::RSA_PSS_2048_8192_SHA512_LEGACY_KEY)
804        }
805        SignatureScheme::ED25519 => Ok(SignatureAlgorithm::ED25519),
806        // In TLS 1.2, ECDSA schemes specify only the hash — the curve
807        // comes from the server certificate's public key.
808        SignatureScheme::ECDSA_NISTP256_SHA256 => {
809            let curve_oid = extract_ec_curve_oid(certs)?;
810            match curve_oid {
811                oid if oid == rfc5912::SECP_256_R_1 => {
812                    Ok(SignatureAlgorithm::ECDSA_NISTP256_SHA256)
813                }
814                oid if oid == rfc5912::SECP_384_R_1 => {
815                    Ok(SignatureAlgorithm::ECDSA_NISTP384_SHA256)
816                }
817                _ => Err(TlsTranscriptError::parse(format!(
818                    "unsupported EC curve: {curve_oid}"
819                ))),
820            }
821        }
822        SignatureScheme::ECDSA_NISTP384_SHA384 => {
823            let curve_oid = extract_ec_curve_oid(certs)?;
824            match curve_oid {
825                oid if oid == rfc5912::SECP_256_R_1 => {
826                    Ok(SignatureAlgorithm::ECDSA_NISTP256_SHA384)
827                }
828                oid if oid == rfc5912::SECP_384_R_1 => {
829                    Ok(SignatureAlgorithm::ECDSA_NISTP384_SHA384)
830                }
831                _ => Err(TlsTranscriptError::parse(format!(
832                    "unsupported EC curve: {curve_oid}"
833                ))),
834            }
835        }
836        _ => Err(TlsTranscriptError::parse(format!(
837            "unsupported signature scheme: {scheme:?}"
838        ))),
839    }
840}

842/// Extract the EC curve OID from the end-entity certificate's SPKI.
843fn extract_ec_curve_oid(certs: &[CertificateDer]) -> Result<ObjectIdentifier, TlsTranscriptError> {
844    let ee_cert = certs
845        .first()
846        .ok_or_else(|| TlsTranscriptError::parse("missing end-entity certificate"))?;
847
848    let cert = pki_types::CertificateDer::from(ee_cert.0.as_slice());
849    let ee = webpki::EndEntityCert::try_from(&cert)
850        .map_err(|e| TlsTranscriptError::parse(format!("invalid end-entity certificate: {e}")))?;
851    let spki_der = ee.subject_public_key_info();
852    let spki = spki::SubjectPublicKeyInfoRef::from_der(spki_der.as_ref())
853        .map_err(|e| TlsTranscriptError::parse(format!("invalid SPKI: {e}")))?;
854    spki.algorithm
855        .parameters
856        .ok_or_else(|| TlsTranscriptError::parse("missing EC curve parameters in SPKI"))?
857        .decode_as::<ObjectIdentifier>()
858        .map_err(|e| TlsTranscriptError::parse(format!("failed to decode EC curve OID: {e}")))
859}

861/// Parses the verify data record without decrypting the plaintext.
862fn parse_verify_data_record(records: &[OpaqueMessage]) -> Result<Record, TlsTranscriptError> {
863    let ccs = find_ccs(records)
864        .ok_or_else(|| TlsTranscriptError::parse("missing ChangeCipherSpec record"))?;
865
866    let raw_finished = records
867        .get(ccs + 1)
868        .ok_or_else(|| TlsTranscriptError::parse("missing Finished record after CCS"))?;
869
870    split_into_record(0, &raw_finished.payload.0, raw_finished.typ)
871}

873/// Parse application data records after the CCS + Finished boundary.
874fn parse_app_records(
875    records: &[OpaqueMessage],
876    app_data: &[u8],
877) -> Result<Vec<Record>, TlsTranscriptError> {
878    let mut consumed = 0;
879    let mut parsed = Vec::new();
880
881    let Some(start) = find_ccs(records) else {
882        return Ok(parsed);
883    };
884
885    // Skip CCS and the Finished record.
886    for (seq, record) in (1u64..).zip(records.iter().skip(start + 2)) {
887        let mut rec = split_into_record(seq, &record.payload.0, record.typ)?;
888
889        if rec.typ == ContentType::ApplicationData {
890            if !app_data.is_empty() {
891                let cipher_len = rec.ciphertext.len();
892                if app_data[consumed..].len() >= cipher_len {
893                    rec.plaintext = Some(app_data[consumed..consumed + cipher_len].to_vec());
894                    consumed += cipher_len;
895                } else {
896                    return Err(TlsTranscriptError::parse(
897                        "insufficient plaintext application data",
898                    ));
899                }
900            }
901            parsed.push(rec);
902        }
903    }
904
905    Ok(parsed)
906}
