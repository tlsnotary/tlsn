41/// A prover instance.
42#[derive(Debug)]
43pub struct Prover<T: state::ProverState = state::Initialized> {
44    config: ProverConfig,
45    span: Span,

49impl Prover<state::Initialized> {
50    /// Creates a new prover.
51    ///
52    /// # Arguments
53    ///
54    /// * `config` - The configuration for the prover.
55    pub fn new(config: ProverConfig) -> Self {
56        let span = info_span!("prover");
57        Self {
58            config,
59            span,
60            state: state::Initialized,
61        }
62    }
63
64    /// Sets up the prover.
65    ///
66    /// This performs all MPC setup prior to establishing the connection to the
67    /// application server.
68    ///
69    /// # Arguments
70    ///
71    /// * `socket` - The socket to the TLS verifier.
72    #[instrument(parent = &self.span, level = "debug", skip_all, err)]
73    pub async fn setup<S: AsyncWrite + AsyncRead + Send + Unpin + 'static>(
74        self,
75        socket: S,
76    ) -> Result<Prover<state::Setup>, ProverError> {
77        let (mut mux_fut, mux_ctrl) = attach_mux(socket, Role::Prover);
78        let mut mt = build_mt_context(mux_ctrl.clone());
79        let mut ctx = mux_fut.poll_with(mt.new_context()).await?;
80
81        // Sends protocol configuration to verifier for compatibility check.
82        mux_fut
83            .poll_with(async {
84                ctx.io_mut()
85                    .send(SetupRequest {
86                        config: self.config.protocol_config().clone(),
87                        version: crate::config::VERSION.clone(),
88                    })
89                    .await?;
90
91                ctx.io_mut()
92                    .expect_next::<Response>()
93                    .await?
94                    .result
95                    .map_err(ProverError::from)
96            })
97            .await?;
98
99        let ProverDeps { vm, mut mpc_tls } =
100            build_prover_deps(self.config.build_mpc_tls_config(), ctx);
101
102        // Allocate resources for MPC-TLS in the VM.
103        let mut keys = mpc_tls.alloc()?;
104        let vm_lock = vm.try_lock().expect("VM is not locked");
105        translate_keys(&mut keys, &vm_lock);
106        drop(vm_lock);
107
108        debug!("setting up mpc-tls");
109
110        mux_fut.poll_with(mpc_tls.preprocess()).await?;
111
112        debug!("mpc-tls setup complete");
113
114        Ok(Prover {
115            config: self.config,
116            span: self.span,
117            state: state::Setup {
118                mux_ctrl,
119                mux_fut,
120                mpc_tls,
121                keys,
122                vm,
123            },
124        })
125    }
126}

128impl Prover<state::Setup> {
129    /// Connects to the server using the provided socket.
130    ///
131    /// Returns a handle to the TLS connection, a future which returns the
132    /// prover once the connection is closed.
133    ///
134    /// # Arguments
135    ///
136    /// * `socket` - The socket to the server.
137    #[instrument(parent = &self.span, level = "debug", skip_all, err)]
138    pub async fn connect<S: AsyncWrite + AsyncRead + Send + Unpin + 'static>(
139        self,
140        socket: S,
141    ) -> Result<(TlsConnection, ProverFuture), ProverError> {
142        let state::Setup {
143            mux_ctrl,
144            mut mux_fut,
145            mpc_tls,
146            keys,
147            vm,
148            ..
149        } = self.state;
150
151        let (mpc_ctrl, mpc_fut) = mpc_tls.run();
152
153        let ServerName::Dns(server_name) = self.config.server_name();
154        let server_name =
155            TlsServerName::try_from(server_name.as_ref()).expect("name was validated");
156
157        let root_store = if let Some(root_store) = self.config.tls_config().root_store() {
158            let roots = root_store
159                .roots
160                .iter()
161                .map(|cert| {
162                    let der = CertificateDer::from_slice(&cert.0);
163                    anchor_from_trusted_cert(&der)
164                        .map(|anchor| anchor.to_owned())
165                        .map_err(ProverError::config)
166                })
167                .collect::<Result<Vec<_>, _>>()?;
168            tls_client::RootCertStore { roots }
169        } else {
170            tls_client::RootCertStore {
171                roots: webpki_roots::TLS_SERVER_ROOTS.to_vec(),
172            }
173        };
174
175        let config = tls_client::ClientConfig::builder()
176            .with_safe_defaults()
177            .with_root_certificates(root_store);
178
179        let config = if let Some((cert, key)) = self.config.tls_config().client_auth() {
180            config
181                .with_single_cert(
182                    cert.iter()
183                        .map(|cert| tls_client::Certificate(cert.0.clone()))
184                        .collect(),
185                    tls_client::PrivateKey(key.0.clone()),
186                )
187                .map_err(ProverError::config)?
188        } else {
189            config.with_no_client_auth()
190        };
191
192        let client =
193            ClientConnection::new(Arc::new(config), Box::new(mpc_ctrl.clone()), server_name)
194                .map_err(ProverError::config)?;
195
196        let (conn, conn_fut) = bind_client(socket, client);
197
198        let fut = Box::pin({
199            let span = self.span.clone();
200            let mpc_ctrl = mpc_ctrl.clone();
201            async move {
202                let conn_fut = async {
203                    mux_fut
204                        .poll_with(conn_fut.map_err(ProverError::from))
205                        .await?;
206
207                    mpc_ctrl.stop().await?;
208
209                    Ok::<_, ProverError>(())
210                };
211
212                info!("starting MPC-TLS");
213
214                let (_, (mut ctx, tls_transcript)) = futures::try_join!(
215                    conn_fut,
216                    mpc_fut.in_current_span().map_err(ProverError::from)
217                )?;
218
219                info!("finished MPC-TLS");
220
221                {
222                    let mut vm = vm.try_lock().expect("VM should not be locked");
223
224                    debug!("finalizing mpc");
225
226                    // Finalize DEAP.
227                    mux_fut
228                        .poll_with(vm.finalize(&mut ctx))
229                        .await
230                        .map_err(ProverError::mpc)?;
231
232                    debug!("mpc finalized");
233                }
234
235                // Pull out ZK VM.
236                let (_, mut vm) = Arc::into_inner(vm)
237                    .expect("vm should have only 1 reference")
238                    .into_inner()
239                    .into_inner();
240
241                // Prove tag verification of received records.
242                // The prover drops the proof output.
243                let _ = verify_tags(
244                    &mut vm,
245                    (keys.server_write_key, keys.server_write_iv),
246                    keys.server_write_mac_key,
247                    *tls_transcript.version(),
248                    tls_transcript.recv().to_vec(),
249                )
250                .map_err(ProverError::zk)?;
251
252                mux_fut
253                    .poll_with(vm.execute_all(&mut ctx).map_err(ProverError::zk))
254                    .await?;
255
256                let transcript = tls_transcript
257                    .to_transcript()
258                    .expect("transcript is complete");
259
260                Ok(Prover {
261                    config: self.config,
262                    span: self.span,
263                    state: state::Committed {
264                        mux_ctrl,
265                        mux_fut,
266                        ctx,
267                        vm,
268                        keys,
269                        tls_transcript,
270                        transcript,
271                    },
272                })
273            }
274            .instrument(span)
275        });
276
277        Ok((
278            conn,
279            ProverFuture {
280                fut,
281                ctrl: ProverControl { mpc_ctrl },
282            },
283        ))
284    }
285}

287impl Prover<state::Committed> {
288    /// Returns the TLS transcript.
289    pub fn tls_transcript(&self) -> &TlsTranscript {
290        &self.state.tls_transcript
291    }
292
293    /// Returns the transcript.
294    pub fn transcript(&self) -> &Transcript {
295        &self.state.transcript
296    }
297
298    /// Proves information to the verifier.
299    ///
300    /// # Arguments
301    ///
302    /// * `config` - The disclosure configuration.
303    #[instrument(parent = &self.span, level = "info", skip_all, err)]
304    pub async fn prove(&mut self, config: &ProveConfig) -> Result<ProverOutput, ProverError> {
305        let state::Committed {
306            mux_fut,
307            ctx,
308            vm,
309            keys,
310            tls_transcript,
311            transcript,
312            ..
313        } = &mut self.state;
314
315        let request = ProveRequest {
316            handshake: config.server_identity().then(|| {
317                (
318                    self.config.server_name().clone(),
319                    HandshakeData {
320                        certs: tls_transcript
321                            .server_cert_chain()
322                            .expect("server cert chain is present")
323                            .to_vec(),
324                        sig: tls_transcript
325                            .server_signature()
326                            .expect("server signature is present")
327                            .clone(),
328                        binding: tls_transcript.certificate_binding().clone(),
329                    },
330                )
331            }),
332            transcript: config
333                .reveal()
334                .map(|(sent, recv)| transcript.to_partial(sent.clone(), recv.clone())),
335            transcript_commit: config.transcript_commit().map(|config| config.to_request()),
336        };
337
338        let output = mux_fut
339            .poll_with(async {
340                ctx.io_mut()
341                    .send(request)
342                    .await
343                    .map_err(ProverError::from)?;
344
345                ctx.io_mut().expect_next::<Response>().await?.result?;
346
347                prove::prove(ctx, vm, keys, transcript, tls_transcript, config).await
348            })
349            .await?;
350
351        Ok(output)
352    }
353
354    /// Closes the connection with the verifier.
355    #[instrument(parent = &self.span, level = "info", skip_all, err)]
356    pub async fn close(self) -> Result<(), ProverError> {
357        let state::Committed {
358            mux_ctrl, mux_fut, ..
359        } = self.state;
360
361        // Wait for the verifier to correctly close the connection.
362        if !mux_fut.is_complete() {
363            mux_ctrl.close();
364            mux_fut.await?;
365        }
366
367        Ok(())
368    }
369}

371/// A controller for the prover.
372#[derive(Clone)]
373pub struct ProverControl {
374    mpc_ctrl: LeaderCtrl,
375}
376
377impl ProverControl {
378    /// Defers decryption of data from the server until the server has closed
379    /// the connection.
380    ///
381    /// This is a performance optimization which will significantly reduce the
382    /// amount of upload bandwidth used by the prover.
383    ///
384    /// # Notes
385    ///
386    /// * The prover may need to close the connection to the server in order for
387    ///   it to close the connection on its end. If neither the prover or server
388    ///   close the connection this will cause a deadlock.
389    pub async fn defer_decryption(&self) -> Result<(), ProverError> {
390        self.mpc_ctrl
391            .defer_decryption()
392            .await
393            .map_err(ProverError::from)
394    }
