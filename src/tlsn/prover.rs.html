52    /// Creates a new prover.
53    ///
54    /// # Arguments
55    ///
56    /// * `config` - The configuration for the prover.
57    pub fn new(config: ProverConfig) -> Self {
58        let span = info_span!("prover");
59        Self {
60            config,
61            span,
62            state: state::Initialized,
63        }
64    }

66    /// Starts the TLS commitment protocol.
67    ///
68    /// This initiates the TLS commitment protocol, including performing any
69    /// necessary preprocessing operations.
70    ///
71    /// # Arguments
72    ///
73    /// * `config` - The TLS commitment configuration.
74    /// * `socket` - The socket to the TLS verifier.
75    #[instrument(parent = &self.span, level = "debug", skip_all, err)]
76    pub async fn commit<S: AsyncWrite + AsyncRead + Send + Unpin + 'static>(
77        self,
78        config: TlsCommitConfig,
79        socket: S,
80    ) -> Result<Prover<state::CommitAccepted>, ProverError> {
81        let (mut mux_fut, mux_ctrl) = attach_mux(socket, Role::Prover);
82        let mut mt = build_mt_context(mux_ctrl.clone());
83        let mut ctx = mux_fut.poll_with(mt.new_context()).await?;
84
85        // Sends protocol configuration to verifier for compatibility check.
86        mux_fut
87            .poll_with(async {
88                ctx.io_mut()
89                    .send(TlsCommitRequestMsg {
90                        request: config.to_request(),
91                        version: crate::VERSION.clone(),
92                    })
93                    .await?;
94
95                ctx.io_mut()
96                    .expect_next::<Response>()
97                    .await?
98                    .result
99                    .map_err(ProverError::from)
100            })
101            .await?;
102
103        let TlsCommitProtocolConfig::Mpc(mpc_tls_config) = config.protocol().clone() else {
104            unreachable!("only MPC TLS is supported");
105        };
106
107        let ProverDeps { vm, mut mpc_tls } = build_prover_deps(mpc_tls_config, ctx);
108
109        // Allocate resources for MPC-TLS in the VM.
110        let mut keys = mpc_tls.alloc()?;
111        let vm_lock = vm.try_lock().expect("VM is not locked");
112        translate_keys(&mut keys, &vm_lock);
113        drop(vm_lock);
114
115        debug!("setting up mpc-tls");
116
117        mux_fut.poll_with(mpc_tls.preprocess()).await?;
118
119        debug!("mpc-tls setup complete");
120
121        Ok(Prover {
122            config: self.config,
123            span: self.span,
124            state: state::CommitAccepted {
125                mux_ctrl,
126                mux_fut,
127                mpc_tls,
128                keys,
129                vm,
130            },
131        })
132    }
133}

135impl Prover<state::CommitAccepted> {
136    /// Connects to the server using the provided socket.
137    ///
138    /// Returns a handle to the TLS connection, a future which returns the
139    /// prover once the connection is closed and the TLS transcript is
140    /// committed.
141    ///
142    /// # Arguments
143    ///
144    /// * `config` - The TLS client configuration.
145    /// * `socket` - The socket to the server.
146    #[instrument(parent = &self.span, level = "debug", skip_all, err)]
147    pub async fn connect<S: AsyncWrite + AsyncRead + Send + Unpin + 'static>(
148        self,
149        config: TlsClientConfig,
150        socket: S,
151    ) -> Result<(TlsConnection, ProverFuture), ProverError> {
152        let state::CommitAccepted {
153            mux_ctrl,
154            mut mux_fut,
155            mpc_tls,
156            keys,
157            vm,
158            ..
159        } = self.state;
160
161        let (mpc_ctrl, mpc_fut) = mpc_tls.run();
162
163        let ServerName::Dns(server_name) = config.server_name();
164        let server_name =
165            TlsServerName::try_from(server_name.as_ref()).expect("name was validated");
166
167        let root_store = tls_client::RootCertStore {
168            roots: config
169                .root_store()
170                .roots
171                .iter()
172                .map(|cert| {
173                    let der = CertificateDer::from_slice(&cert.0);
174                    anchor_from_trusted_cert(&der)
175                        .map(|anchor| anchor.to_owned())
176                        .map_err(ProverError::config)
177                })
178                .collect::<Result<Vec<_>, _>>()?,
179        };
180
181        let rustls_config = tls_client::ClientConfig::builder()
182            .with_safe_defaults()
183            .with_root_certificates(root_store);
184
185        let rustls_config = if let Some((cert, key)) = config.client_auth() {
186            rustls_config
187                .with_single_cert(
188                    cert.iter()
189                        .map(|cert| tls_client::Certificate(cert.0.clone()))
190                        .collect(),
191                    tls_client::PrivateKey(key.0.clone()),
192                )
193                .map_err(ProverError::config)?
194        } else {
195            rustls_config.with_no_client_auth()
196        };
197
198        let client = ClientConnection::new(
199            Arc::new(rustls_config),
200            Box::new(mpc_ctrl.clone()),
201            server_name,
202        )
203        .map_err(ProverError::config)?;
204
205        let (conn, conn_fut) = bind_client(socket, client);
206
207        let fut = Box::pin({
208            let span = self.span.clone();
209            let mpc_ctrl = mpc_ctrl.clone();
210            async move {
211                let conn_fut = async {
212                    mux_fut
213                        .poll_with(conn_fut.map_err(ProverError::from))
214                        .await?;
215
216                    mpc_ctrl.stop().await?;
217
218                    Ok::<_, ProverError>(())
219                };
220
221                info!("starting MPC-TLS");
222
223                let (_, (mut ctx, tls_transcript)) = futures::try_join!(
224                    conn_fut,
225                    mpc_fut.in_current_span().map_err(ProverError::from)
226                )?;
227
228                info!("finished MPC-TLS");
229
230                {
231                    let mut vm = vm.try_lock().expect("VM should not be locked");
232
233                    debug!("finalizing mpc");
234
235                    // Finalize DEAP.
236                    mux_fut
237                        .poll_with(vm.finalize(&mut ctx))
238                        .await
239                        .map_err(ProverError::mpc)?;
240
241                    debug!("mpc finalized");
242                }
243
244                // Pull out ZK VM.
245                let (_, mut vm) = Arc::into_inner(vm)
246                    .expect("vm should have only 1 reference")
247                    .into_inner()
248                    .into_inner();
249
250                // Prove tag verification of received records.
251                // The prover drops the proof output.
252                let _ = verify_tags(
253                    &mut vm,
254                    (keys.server_write_key, keys.server_write_iv),
255                    keys.server_write_mac_key,
256                    *tls_transcript.version(),
257                    tls_transcript.recv().to_vec(),
258                )
259                .map_err(ProverError::zk)?;
260
261                mux_fut
262                    .poll_with(vm.execute_all(&mut ctx).map_err(ProverError::zk))
263                    .await?;
264
265                let transcript = tls_transcript
266                    .to_transcript()
267                    .expect("transcript is complete");
268
269                Ok(Prover {
270                    config: self.config,
271                    span: self.span,
272                    state: state::Committed {
273                        mux_ctrl,
274                        mux_fut,
275                        ctx,
276                        vm,
277                        server_name: config.server_name().clone(),
278                        keys,
279                        tls_transcript,
280                        transcript,
281                    },
282                })
283            }
284            .instrument(span)
285        });
286
287        Ok((
288            conn,
289            ProverFuture {
290                fut,
291                ctrl: ProverControl { mpc_ctrl },
292            },
293        ))
294    }
295}

297impl Prover<state::Committed> {
298    /// Returns the TLS transcript.
299    pub fn tls_transcript(&self) -> &TlsTranscript {
300        &self.state.tls_transcript
301    }
302
303    /// Returns the transcript.
304    pub fn transcript(&self) -> &Transcript {
305        &self.state.transcript
306    }
307
308    /// Proves information to the verifier.
309    ///
310    /// # Arguments
311    ///
312    /// * `config` - The disclosure configuration.
313    #[instrument(parent = &self.span, level = "info", skip_all, err)]
314    pub async fn prove(&mut self, config: &ProveConfig) -> Result<ProverOutput, ProverError> {
315        let state::Committed {
316            mux_fut,
317            ctx,
318            vm,
319            keys,
320            server_name,
321            tls_transcript,
322            transcript,
323            ..
324        } = &mut self.state;
325
326        let handshake = config.server_identity().then(|| {
327            (
328                server_name.clone(),
329                HandshakeData {
330                    certs: tls_transcript
331                        .server_cert_chain()
332                        .expect("server cert chain is present")
333                        .to_vec(),
334                    sig: tls_transcript
335                        .server_signature()
336                        .expect("server signature is present")
337                        .clone(),
338                    binding: tls_transcript.certificate_binding().clone(),
339                },
340            )
341        });
342
343        let partial_transcript = config
344            .reveal()
345            .map(|(sent, recv)| transcript.to_partial(sent.clone(), recv.clone()));
346
347        let msg = ProveRequestMsg {
348            request: config.to_request(),
349            handshake,
350            transcript: partial_transcript,
351        };
352
353        let output = mux_fut
354            .poll_with(async {
355                ctx.io_mut().send(msg).await.map_err(ProverError::from)?;
356
357                ctx.io_mut().expect_next::<Response>().await?.result?;
358
359                prove::prove(ctx, vm, keys, transcript, tls_transcript, config).await
360            })
361            .await?;
362
363        Ok(output)
364    }
365
366    /// Closes the connection with the verifier.
367    #[instrument(parent = &self.span, level = "info", skip_all, err)]
368    pub async fn close(self) -> Result<(), ProverError> {
369        let state::Committed {
370            mux_ctrl, mux_fut, ..
371        } = self.state;
372
373        // Wait for the verifier to correctly close the connection.
374        if !mux_fut.is_complete() {
375            mux_ctrl.close();
376            mux_fut.await?;
377        }
378
379        Ok(())
380    }
381}

383/// A controller for the prover.
384#[derive(Clone)]
385pub struct ProverControl {
386    mpc_ctrl: LeaderCtrl,
387}
388
389impl ProverControl {
390    /// Defers decryption of data from the server until the server has closed
391    /// the connection.
392    ///
393    /// This is a performance optimization which will significantly reduce the
394    /// amount of upload bandwidth used by the prover.
395    ///
396    /// # Notes
397    ///
398    /// * The prover may need to close the connection to the server in order for
399    ///   it to close the connection on its end. If neither the prover or server
400    ///   close the connection this will cause a deadlock.
401    pub async fn defer_decryption(&self) -> Result<(), ProverError> {
402        self.mpc_ctrl
403            .defer_decryption()
404            .await
405            .map_err(ProverError::from)
406    }
