14/// Multiplexer supporting unique deterministic stream IDs.

16/// Multiplexer controller providing streams.

19/// Multiplexer future which must be polled for the muxer to make progress.
20pub(crate) struct MuxFuture(
21    Box<dyn FusedFuture<Output = Result<(), yamux::ConnectionError>> + Send + Unpin>,
22);
23
24impl MuxFuture {
25    /// Returns true if the muxer is complete.
26    pub(crate) fn is_complete(&self) -> bool {
27        self.0.is_terminated()
28    }
29
30    /// Awaits a future, polling the muxer future concurrently.
31    pub(crate) async fn poll_with<F, R>(&mut self, fut: F) -> R
32    where
33        F: Future<Output = R>,
34    {
35        let mut fut = Box::pin(fut.fuse());
36        // Poll the future concurrently with the muxer future.
37        // If the muxer returns an error, continue polling the future
38        // until it completes.
39        loop {
40            futures::select! {
41                res = fut => return res,
42                res = &mut self.0 => if let Err(e) = res {
43                    error!("mux error: {:?}", e);
44                },
45            }
46        }
47    }
48}
49
50impl Future for MuxFuture {
51    type Output = Result<(), yamux::ConnectionError>;
52
53    fn poll(
54        mut self: std::pin::Pin<&mut Self>,
55        cx: &mut std::task::Context<'_>,
56    ) -> std::task::Poll<Self::Output> {
57        self.0.as_mut().poll_unpin(cx)
58    }
59}

61/// Attaches a multiplexer to the provided socket.
62///
63/// Returns the multiplexer and a controller for creating streams with a codec
64/// attached.
65///
66/// # Arguments
67///
68/// * `socket` - The socket to attach the multiplexer to.
69/// * `role` - The role of the party using the multiplexer.
70pub(crate) fn attach_mux<T: AsyncWrite + AsyncRead + Send + Unpin + 'static>(
71    socket: T,
72    role: Role,
73) -> (MuxFuture, MuxControl) {
74    let mut mux_config = yamux::Config::default();
75    mux_config.set_max_num_streams(36);
76
77    let mux_role = match role {
78        Role::Prover => yamux::Mode::Client,
79        Role::Verifier => yamux::Mode::Server,
80    };
81
82    let mux = Mux::new(socket, mux_config, mux_role);
83    let ctrl = mux.control();
84
85    if let Role::Prover = role {
86        ctrl.alloc(32);
87    }
88
89    (MuxFuture(Box::new(mux.into_future().fuse())), ctrl)
90}
