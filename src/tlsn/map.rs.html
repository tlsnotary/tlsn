6#[derive(Debug, Clone, PartialEq)]
7pub(crate) struct RangeMap<T> {
8    map: Vec<(usize, T)>,
9}
10
11impl<T> Default for RangeMap<T>
12where
13    T: Item,
14{
15    fn default() -> Self {
16        Self { map: Vec::new() }
17    }
18}

20impl<T> RangeMap<T>
21where
22    T: Item,
23{
24    /// Returns `true` if the map is empty.
25    pub(crate) fn is_empty(&self) -> bool {
26        self.map.is_empty()
27    }
28
29    /// Returns the keys of the map.
30    pub(crate) fn keys(&self) -> impl Iterator<Item = Range<usize>> {
31        self.map
32            .iter()
33            .map(|(idx, item)| *idx..*idx + item.length())
34    }
35
36    pub(crate) fn iter(&self) -> impl Iterator<Item = (Range<usize>, &T)> {
37        self.map
38            .iter()
39            .map(|(idx, item)| (*idx..*idx + item.length(), item))
40    }
41
42    pub(crate) fn get(&self, range: Range<usize>) -> Option<T::Slice<'_>> {
43        if range.start >= range.end {
44            return None;
45        }
46
47        // Find the item with the greatest start index <= range.start
48        let pos = match self.map.binary_search_by(|(idx, _)| idx.cmp(&range.start)) {
49            Ok(i) => i,
50            Err(0) => return None,
51            Err(i) => i - 1,
52        };
53
54        let (base, item) = &self.map[pos];
55
56        item.slice(range.start - *base..range.end - *base)
57    }
58
59    pub(crate) fn index(&self, idx: &RangeSet<usize>) -> Option<Self> {
60        let mut map = Vec::new();
61        for idx in idx.iter() {
62            let pos = match self.map.binary_search_by(|(base, _)| base.cmp(&idx.start)) {
63                Ok(i) => i,
64                Err(0) => return None,
65                Err(i) => i - 1,
66            };
67
68            let (base, item) = self.map.get(pos)?;
69            if idx.start < *base || idx.end > *base + item.length() {
70                return None;
71            }
72
73            let start = idx.start - *base;
74            let end = start + idx.len();
75
76            map.push((
77                idx.start,
78                item.slice(start..end)
79                    .expect("slice length is checked")
80                    .into(),
81            ));
82        }
83
84        Some(Self { map })
85    }
86}

88impl<T> FromIterator<(usize, T)> for RangeMap<T>
89where
90    T: Item,
91{
92    fn from_iter<I: IntoIterator<Item = (usize, T)>>(items: I) -> Self {
93        let mut pos = 0;
94        let mut map = Vec::new();
95        for (idx, item) in items {
96            assert!(
97                idx >= pos,
98                "items must be sorted by index and non-overlapping"
99            );
100
101            pos = idx + item.length();
102            map.push((idx, item));
103        }
104
105        Self { map }
106    }
107}

109pub(crate) trait Item: Sized {
110    type Slice<'a>: Into<Self>
111    where
112        Self: 'a;
113
114    fn length(&self) -> usize;
115
116    fn slice<'a>(&'a self, range: Range<usize>) -> Option<Self::Slice<'a>>;
117}

119impl Item for Vector<U8> {
120    type Slice<'a> = Vector<U8>;
121
122    fn length(&self) -> usize {
123        self.len()
124    }
125
126    fn slice<'a>(&'a self, range: Range<usize>) -> Option<Self::Slice<'a>> {
127        self.get(range)
128    }
129}

143            if range.end > self.end - self.start {
144                return None;
145            }

148        }
