26pub(crate) async fn verify<T: Vm<Binary> + KeyStore + Send + Sync>(
27    ctx: &mut Context,
28    vm: &mut T,
29    keys: &SessionKeys,
30    cert_verifier: &ServerCertVerifier,
31    tls_transcript: &TlsTranscript,
32    request: ProveRequest,
33) -> Result<VerifierOutput, VerifierError> {
34    let ProveRequest {
35        handshake,
36        transcript,
37        transcript_commit,
38    } = request;
39
40    let ciphertext_sent = collect_ciphertext(tls_transcript.sent());
41    let ciphertext_recv = collect_ciphertext(tls_transcript.recv());
42
43    let has_reveal = transcript.is_some();
44    let transcript = if let Some(transcript) = transcript {
45        if transcript.len_sent() != ciphertext_sent.len()
46            || transcript.len_received() != ciphertext_recv.len()
47        {
48            return Err(VerifierError::verify(
49                "prover sent transcript with incorrect length",
50            ));
51        }
52
53        transcript
54    } else {
55        PartialTranscript::new(ciphertext_sent.len(), ciphertext_recv.len())
56    };
57
58    let server_name = if let Some((name, cert_data)) = handshake {
59        cert_data
60            .verify(
61                cert_verifier,
62                tls_transcript.time(),
63                tls_transcript.server_ephemeral_key(),
64                &name,
65            )
66            .map_err(VerifierError::verify)?;
67
68        Some(name)
69    } else {
70        None
71    };
72
73    let (mut commit_sent, mut commit_recv) = (RangeSet::default(), RangeSet::default());
74    if let Some(commit_config) = transcript_commit.as_ref() {
75        commit_config
76            .iter_hash()
77            .for_each(|(direction, idx, _)| match direction {
78                Direction::Sent => commit_sent.union_mut(idx),
79                Direction::Received => commit_recv.union_mut(idx),
80            });
81
82        if let Some((sent, recv)) = commit_config.encoding() {
83            commit_sent.union_mut(sent);
84            commit_recv.union_mut(recv);
85        }
86    }
87
88    let (sent_refs, sent_proof) = verify_plaintext(
89        vm,
90        keys.client_write_key,
91        keys.client_write_iv,
92        transcript.sent_unsafe(),
93        &ciphertext_sent,
94        tls_transcript
95            .sent()
96            .iter()
97            .filter(|record| record.typ == ContentType::ApplicationData),
98        transcript.sent_authed(),
99        &commit_sent,
100    )
101    .map_err(VerifierError::zk)?;
102    let (recv_refs, recv_proof) = verify_plaintext(
103        vm,
104        keys.server_write_key,
105        keys.server_write_iv,
106        transcript.received_unsafe(),
107        &ciphertext_recv,
108        tls_transcript
109            .recv()
110            .iter()
111            .filter(|record| record.typ == ContentType::ApplicationData),
112        transcript.received_authed(),
113        &commit_recv,
114    )
115    .map_err(VerifierError::zk)?;
116
117    let transcript_refs = TranscriptRefs {
118        sent: sent_refs,
119        recv: recv_refs,
120    };
121
122    let mut transcript_commitments = Vec::new();
123    let mut hash_commitments = None;
124    if let Some(commit_config) = transcript_commit.as_ref()
125        && commit_config.has_hash()
126    {
127        hash_commitments = Some(
128            verify_hash(vm, &transcript_refs, commit_config.iter_hash().cloned())
129                .map_err(VerifierError::verify)?,
130        );
131    }
132
133    vm.execute_all(ctx).await.map_err(VerifierError::zk)?;
134
135    sent_proof.verify().map_err(VerifierError::verify)?;
136    recv_proof.verify().map_err(VerifierError::verify)?;
137
138    let mut encoder_secret = None;
139    if let Some(commit_config) = transcript_commit
140        && let Some((sent, recv)) = commit_config.encoding()
141    {
142        let sent_map = transcript_refs
143            .sent
144            .index(sent)
145            .expect("ranges were authenticated");
146        let recv_map = transcript_refs
147            .recv
148            .index(recv)
149            .expect("ranges were authenticated");
150
151        let (secret, commitment) = encoding::transfer(ctx, vm, &sent_map, &recv_map).await?;
152        encoder_secret = Some(secret);
153        transcript_commitments.push(TranscriptCommitment::Encoding(commitment));
154    }
155
156    if let Some(hash_commitments) = hash_commitments {
157        for commitment in hash_commitments.try_recv().map_err(VerifierError::verify)? {
158            transcript_commitments.push(TranscriptCommitment::Hash(commitment));
159        }
160    }
161
162    Ok(VerifierOutput {
163        server_name,
164        transcript: has_reveal.then_some(transcript),
165        encoder_secret,
166        transcript_commitments,
167    })
168}

170fn collect_ciphertext<'a>(records: impl IntoIterator<Item = &'a Record>) -> Vec<u8> {
171    let mut ciphertext = Vec::new();
172    records
173        .into_iter()
174        .filter(|record| record.typ == ContentType::ApplicationData)
175        .for_each(|record| {
176            ciphertext.extend_from_slice(&record.ciphertext);
177        });
178    ciphertext
179}
