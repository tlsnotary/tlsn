1use std::fmt::{Debug, Formatter, Result};

3use mpc_tls::Config;
4use serde::{Deserialize, Serialize};
5use tlsn_core::webpki::RootCertStore;

7use crate::config::{NetworkSetting, ProtocolConfig, ProtocolConfigValidator};

9/// Configuration for the [`Verifier`](crate::tls::Verifier).
10#[allow(missing_docs)]
11#[derive(derive_builder::Builder, Serialize, Deserialize)]
12#[builder(pattern = "owned")]
13pub struct VerifierConfig {
14    protocol_config_validator: ProtocolConfigValidator,
15    #[builder(setter(strip_option))]
16    root_store: Option<RootCertStore>,
17}
18
19impl Debug for VerifierConfig {
20    fn fmt(&self, f: &mut Formatter<'_>) -> Result {
21        f.debug_struct("VerifierConfig")
22            .field("protocol_config_validator", &self.protocol_config_validator)
23            .finish_non_exhaustive()
24    }
25}
26
27impl VerifierConfig {
28    /// Creates a new configuration builder.
29    pub fn builder() -> VerifierConfigBuilder {
30        VerifierConfigBuilder::default()
31    }
32
33    /// Returns the protocol configuration validator.
34    pub fn protocol_config_validator(&self) -> &ProtocolConfigValidator {
35        &self.protocol_config_validator
36    }
37
38    /// Returns the root certificate store.
39    pub fn root_store(&self) -> Option<&RootCertStore> {
40        self.root_store.as_ref()
41    }
42
43    pub(crate) fn build_mpc_tls_config(&self, protocol_config: &ProtocolConfig) -> Config {
44        let mut builder = Config::builder();
45
46        builder
47            .max_sent(protocol_config.max_sent_data())
48            .max_recv_online(protocol_config.max_recv_data_online())
49            .max_recv(protocol_config.max_recv_data());
50
51        if let Some(max_sent_records) = protocol_config.max_sent_records() {
52            builder.max_sent_records(max_sent_records);
53        }
54
55        if let Some(max_recv_records_online) = protocol_config.max_recv_records_online() {
56            builder.max_recv_records_online(max_recv_records_online);
57        }
58
59        if let NetworkSetting::Latency = protocol_config.network() {
60            builder.low_bandwidth();
61        }
62
63        builder.build().unwrap()
64    }
