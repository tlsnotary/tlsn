1use std::fmt::{Debug, Formatter, Result};

3use mpc_tls::Config;
4use serde::{Deserialize, Serialize};

7use crate::config::{NetworkSetting, ProtocolConfig};

9/// Configuration for the [`Verifier`](crate::tls::Verifier).
10#[allow(missing_docs)]
11#[derive(derive_builder::Builder, Serialize, Deserialize)]
12#[builder(pattern = "owned")]
13pub struct VerifierConfig {
14    #[builder(default, setter(strip_option))]
15    root_store: Option<RootCertStore>,
16}
17
18impl Debug for VerifierConfig {
19    fn fmt(&self, f: &mut Formatter<'_>) -> Result {
20        f.debug_struct("VerifierConfig").finish_non_exhaustive()
21    }
22}
23
24impl VerifierConfig {
25    /// Creates a new configuration builder.
26    pub fn builder() -> VerifierConfigBuilder {
27        VerifierConfigBuilder::default()
28    }
29
30    /// Returns the root certificate store.
31    pub fn root_store(&self) -> Option<&RootCertStore> {
32        self.root_store.as_ref()
33    }
34
35    pub(crate) fn build_mpc_tls_config(&self, protocol_config: &ProtocolConfig) -> Config {
36        let mut builder = Config::builder();
37
38        builder
39            .max_sent(protocol_config.max_sent_data())
40            .max_recv_online(protocol_config.max_recv_data_online())
41            .max_recv(protocol_config.max_recv_data());
42
43        if let Some(max_sent_records) = protocol_config.max_sent_records() {
44            builder.max_sent_records(max_sent_records);
45        }
46
47        if let Some(max_recv_records_online) = protocol_config.max_recv_records_online() {
48            builder.max_recv_records_online(max_recv_records_online);
49        }
50
51        if let NetworkSetting::Latency = protocol_config.network() {
52            builder.low_bandwidth();
53        }
54
55        builder.build().unwrap()
56    }
