57pub(crate) fn build_prover_deps(config: MpcTlsConfig, ctx: Context) -> ProverDeps {
58    let mut rng = rand::rng();
59    let delta = Delta::new(Block::random(&mut rng));
60
61    let base_ot_send = co::Sender::default();
62    let base_ot_recv = co::Receiver::default();
63    let rcot_send = kos::Sender::new(
64        kos::SenderConfig::default(),
65        delta.into_inner(),
66        base_ot_recv,
67    );
68    let rcot_recv = kos::Receiver::new(kos::ReceiverConfig::default(), base_ot_send);
69    let rcot_recv = ferret::Receiver::new(
70        ferret::FerretConfig::builder()
71            .lpn_type(ferret::LpnType::Regular)
72            .build()
73            .expect("ferret config is valid"),
74        Block::random(&mut rng),
75        rcot_recv,
76    );
77
78    let rcot_send = SharedRCOTSender::new(rcot_send);
79    let rcot_recv = SharedRCOTReceiver::new(rcot_recv);
80
81    #[cfg(not(tlsn_insecure))]
82    let mpc = ProverMpc::new(DerandCOTSender::new(rcot_send.clone()), rng.random(), delta);
83    #[cfg(tlsn_insecure)]
84    let mpc = mpz_ideal_vm::IdealVm::new();
85
86    #[cfg(not(tlsn_insecure))]
87    let zk = ProverZk::new(Default::default(), rcot_recv.clone());
88    #[cfg(tlsn_insecure)]
89    let zk = mpz_ideal_vm::IdealVm::new();
90
91    let vm = Arc::new(Mutex::new(Deap::new(tlsn_deap::Role::Leader, mpc, zk)));
92    let mpc_tls = MpcTlsLeader::new(
93        config,
94        ctx,
95        vm.clone(),
96        (rcot_send.clone(), rcot_send.clone(), rcot_send),
97        rcot_recv,
98    );
99
100    ProverDeps { vm, mpc_tls }
101}

108pub(crate) fn build_verifier_deps(config: MpcTlsConfig, ctx: Context) -> VerifierDeps {
109    let mut rng = rand::rng();
110
111    let delta = Delta::random(&mut rng);
112    let base_ot_send = co::Sender::default();
113    let base_ot_recv = co::Receiver::default();
114    let rcot_send = kos::Sender::new(
115        kos::SenderConfig::default(),
116        delta.into_inner(),
117        base_ot_recv,
118    );
119    let rcot_send = ferret::Sender::new(
120        ferret::FerretConfig::builder()
121            .lpn_type(ferret::LpnType::Regular)
122            .build()
123            .expect("ferret config is valid"),
124        Block::random(&mut rng),
125        rcot_send,
126    );
127    let rcot_recv = kos::Receiver::new(kos::ReceiverConfig::default(), base_ot_send);
128
129    let rcot_send = SharedRCOTSender::new(rcot_send);
130    let rcot_recv = SharedRCOTReceiver::new(rcot_recv);
131
132    #[cfg(not(tlsn_insecure))]
133    let mpc = VerifierMpc::new(DerandCOTReceiver::new(rcot_recv.clone()));
134    #[cfg(tlsn_insecure)]
135    let mpc = mpz_ideal_vm::IdealVm::new();
136
137    #[cfg(not(tlsn_insecure))]
138    let zk = VerifierZk::new(Default::default(), delta, rcot_send.clone());
139    #[cfg(tlsn_insecure)]
140    let zk = mpz_ideal_vm::IdealVm::new();
141
142    let vm = Arc::new(Mutex::new(Deap::new(tlsn_deap::Role::Follower, mpc, zk)));
143    let mpc_tls = MpcTlsFollower::new(
144        config,
145        ctx,
146        vm.clone(),
147        rcot_send,
148        (rcot_recv.clone(), rcot_recv.clone(), rcot_recv),
149    );
150
151    VerifierDeps { vm, mpc_tls }
152}

154pub(crate) fn translate_keys<Mpc, Zk>(keys: &mut SessionKeys, vm: &Deap<Mpc, Zk>) {
155    keys.client_write_key = vm
156        .translate(keys.client_write_key)
157        .expect("VM memory should be consistent");
158    keys.client_write_iv = vm
159        .translate(keys.client_write_iv)
160        .expect("VM memory should be consistent");
161    keys.server_write_key = vm
162        .translate(keys.server_write_key)
163        .expect("VM memory should be consistent");
164    keys.server_write_iv = vm
165        .translate(keys.server_write_iv)
166        .expect("VM memory should be consistent");
167    keys.server_write_mac_key = vm
168        .translate(keys.server_write_mac_key)
169        .expect("VM memory should be consistent");
170}
