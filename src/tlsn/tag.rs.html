15/// Proves the verification of tags of the given `records`,
16/// returning a proof.
17///
18/// # Arguments
19///
20/// * `vm` - Virtual machine.
21/// * `key_iv` - Cipher key and IV.
22/// * `mac_key` - MAC key.
23/// * `tls_version` - TLS protocol version.
24/// * `records` - Records for which the verification is to be proven.
25pub(crate) fn verify_tags(
26    vm: &mut dyn Vm<Binary>,
27    key_iv: (Array<U8, 16>, Array<U8, 4>),
28    mac_key: Array<U8, 16>,
29    tls_version: TlsVersion,
30    records: Vec<Record>,
31) -> Result<TagProof, TagProofError> {
32    let mut aes = Aes128::default();
33    aes.set_key(key_iv.0);
34    aes.set_iv(key_iv.1);
35
36    // Compute j0 blocks.
37    let j0s = records
38        .iter()
39        .map(|rec| {
40            let block = aes.alloc_ctr_block(vm).map_err(TagProofError::vm)?;
41
42            let explicit_nonce: [u8; 8] =
43                rec.explicit_nonce
44                    .clone()
45                    .try_into()
46                    .map_err(|explicit_nonce: Vec<_>| ErrorRepr::ExplicitNonceLength {
47                        expected: 8,
48                        actual: explicit_nonce.len(),
49                    })?;
50
51            vm.assign(block.explicit_nonce, explicit_nonce)
52                .map_err(TagProofError::vm)?;
53            vm.commit(block.explicit_nonce).map_err(TagProofError::vm)?;
54
55            // j0's counter is set to 1.
56            vm.assign(block.counter, 1u32.to_be_bytes())
57                .map_err(TagProofError::vm)?;
58            vm.commit(block.counter).map_err(TagProofError::vm)?;
59
60            let j0 = vm.decode(block.output).map_err(TagProofError::vm)?;
61
62            Ok(j0)
63        })
64        .collect::<Result<Vec<_>, TagProofError>>()?;
65
66    let mac_key = vm.decode(mac_key).map_err(TagProofError::vm)?;
67
68    Ok(TagProof {
69        tls_version,
70        j0s,
71        records,
72        mac_key,
73    })
74}

76/// Proof of tag verification.
77#[derive(Debug)]
78#[must_use]
79pub(crate) struct TagProof {
80    tls_version: TlsVersion,
81    /// The j0 block for each record.
82    j0s: Vec<DecodeFutureTyped<BitVec, [u8; 16]>>,
83    records: Vec<Record>,
84    /// The MAC key for tag computation.
85    mac_key: DecodeFutureTyped<BitVec, [u8; 16]>,
86}
87
88impl TagProof {
89    /// Verifies the proof.
90    pub(crate) fn verify(self) -> Result<(), TagProofError> {
91        let Self {
92            tls_version,
93            j0s,
94            mut mac_key,
95            records,
96        } = self;
97
98        let mac_key = mac_key
99            .try_recv()
100            .map_err(TagProofError::vm)?
101            .ok_or_else(|| ErrorRepr::NotDecoded)?;
102
103        let vers = match tls_version {
104            TlsVersion::V1_2 => ProtocolVersion::TLSv1_2,
105            TlsVersion::V1_3 => ProtocolVersion::TLSv1_3,
106        };
107
108        for (mut j0, rec) in j0s.into_iter().zip(records) {
109            let j0 = j0
110                .try_recv()
111                .map_err(TagProofError::vm)?
112                .ok_or_else(|| ErrorRepr::NotDecoded)?;
113
114            let aad = make_tls12_aad(rec.seq, rec.typ.into(), vers, rec.ciphertext.len());
115
116            let ghash_tag = ghash(aad.as_ref(), &rec.ciphertext, &mac_key);
117
118            let record_tag = match rec.tag.as_ref() {
119                Some(tag) => tag,
120                None => {
121                    // This will never happen, since we only call this method
122                    // for proofs where the records' tags are known.
123                    return Err(ErrorRepr::UnknownTag.into());
124                }
125            };
126
127            if *record_tag
128                != ghash_tag
129                    .into_iter()
130                    .zip(j0)
131                    .map(|(a, b)| a ^ b)
132                    .collect::<Vec<_>>()
133            {
134                return Err(ErrorRepr::InvalidTag.into());
135            }
136        }
137
138        Ok(())
139    }
140}

142/// Error for [`J0Proof`].
143#[derive(Debug, thiserror::Error)]
144#[error(transparent)]
145pub(crate) struct TagProofError(#[from] ErrorRepr);
146
147impl TagProofError {
148    fn vm<E>(err: E) -> Self
149    where
150        E: Into<Box<dyn std::error::Error + Send + Sync + 'static>>,
151    {
152        Self(ErrorRepr::Vm(err.into()))
153    }
154}
155
156#[derive(Debug, thiserror::Error)]
157#[error("j0 proof error: {0}")]
158enum ErrorRepr {
159    #[error("value was not decoded")]
160    NotDecoded,
161    #[error("VM error: {0}")]
162    Vm(Box<dyn std::error::Error + Send + Sync + 'static>),
163    #[error("tag does not match expected")]
164    InvalidTag,
165    #[error("tag is not known")]
166    UnknownTag,
167    #[error("invalid explicit nonce length: expected {expected}, got {actual}")]
168    ExplicitNonceLength { expected: usize, actual: usize },
169}
