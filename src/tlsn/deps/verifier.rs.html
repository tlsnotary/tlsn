22cfg_select! {
23    tlsn_insecure => {
24        pub(crate) type VerifierMpc = mpz_ideal_vm::IdealVm;
25        pub(crate) type VerifierZk = mpz_ideal_vm::IdealVm;

39/// Protocol dependencies for Mpc.
40pub(crate) struct VerifierMpcDeps {
41    pub(crate) vm: Arc<Mutex<Deap<VerifierMpc, VerifierZk>>>,
42    pub(crate) mpc_tls: Box<MpcTlsFollower>,
43    pub(crate) keys: Option<SessionKeys>,
44}
45
46impl std::fmt::Debug for VerifierMpcDeps {
47    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
48        f.debug_struct("VerifierMpcDeps").finish_non_exhaustive()
49    }
50}
51
52impl VerifierMpcDeps {
53    pub(crate) fn new(config: &MpcTlsConfig, ctx: Context) -> Self {
54        let mut rng = rand::rng();
55
56        let delta = Delta::random(&mut rng);
57        let base_ot_send = co::Sender::default();
58        let base_ot_recv = co::Receiver::default();
59        let rcot_send = kos::Sender::new(
60            kos::SenderConfig::default(),
61            delta.into_inner(),
62            base_ot_recv,
63        );
64        let rcot_send = ferret::Sender::new(
65            ferret::FerretConfig::builder()
66                .lpn_type(ferret::LpnType::Regular)
67                .build()
68                .expect("ferret config is valid"),
69            Block::random(&mut rng),
70            rcot_send,
71        );
72        let rcot_recv = kos::Receiver::new(kos::ReceiverConfig::default(), base_ot_send);
73
74        let rcot_send = SharedRCOTSender::new(rcot_send);
75        let rcot_recv = SharedRCOTReceiver::new(rcot_recv);
76
77        let mpc = cfg_select! {
78            tlsn_insecure => { mpz_ideal_vm::IdealVm::new() }
79            _ => { VerifierMpc::new(DerandCOTReceiver::new(rcot_recv.clone())) }
80        };
81
82        let zk = cfg_select! {
83            tlsn_insecure => { mpz_ideal_vm::IdealVm::new() }
84            _ => { VerifierZk::new(Default::default(), delta, rcot_send.clone()) }
85        };
86
87        let vm = Arc::new(Mutex::new(Deap::new(tlsn_deap::Role::Follower, mpc, zk)));
88        let mpc_tls = MpcTlsFollower::new(
89            build_mpc_tls_config(config),
90            ctx,
91            vm.clone(),
92            rcot_send,
93            (rcot_recv.clone(), rcot_recv.clone(), rcot_recv),
94        );
95
96        Self {
97            vm,
98            mpc_tls: Box::new(mpc_tls),
99            keys: None,
100        }
101    }
102
103    pub(crate) async fn setup(&mut self) -> Result<(), Error> {
104        let mut keys = self.mpc_tls.alloc().map_err(|e| {
105            Error::internal()
106                .with_msg("commitment protocol failed to allocate mpc-tls resources")
107                .with_source(e)
108        })?;
109        let vm_lock = self.vm.try_lock().expect("VM is not locked");
110        translate_keys(&mut keys, &vm_lock);
111        self.keys = Some(keys);
112
113        drop(vm_lock);
114
115        debug!("setting up mpc-tls");
116        self.mpc_tls.preprocess().await.map_err(|e| {
117            Error::internal()
118                .with_msg("commitment protocol failed during mpc-tls preprocessing")
119                .with_source(e)
120        })?;
121
122        Ok(())
123    }
124}

126/// Protocol dependencies for Proxy.
127pub(crate) struct VerifierProxyDeps {
128    pub(crate) verifier: Box<ProxyVerifier>,
129    pub(crate) id: ThreadId,
130}
131
132impl std::fmt::Debug for VerifierProxyDeps {
133    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
134        f.debug_struct("VerifierProxyDeps").finish_non_exhaustive()
135    }
136}
137
138impl VerifierProxyDeps {
139    pub(crate) fn new(_config: &ProxyTlsConfig, ctx: Context) -> Self {
140        let vm = cfg_select! {
141            tlsn_insecure => { mpz_ideal_vm::IdealVm::new() }
142            _ => {{
143                let mut rng = rand::rng();
144                let delta = Delta::random(&mut rng);
145
146                let base_ot_recv = co::Receiver::default();
147                let rcot_send = kos::Sender::new(
148                    kos::SenderConfig::default(),
149                    delta.into_inner(),
150                    base_ot_recv,
151                );
152                let rcot_send = ferret::Sender::new(
153                    ferret::FerretConfig::builder()
154                        .lpn_type(ferret::LpnType::Regular)
155                        .build()
156                        .expect("ferret config is valid"),
157                    Block::random(&mut rng),
158                    rcot_send,
159                );
160                let rcot_send = SharedRCOTSender::new(rcot_send);
161                VerifierZk::new(Default::default(), delta, rcot_send)
162            }}
163        };
164
165        let prf_config = PrfConfig::new(NetworkMode::Normal, MSMode::Direct);
166        let prf = Prf::new(prf_config);
167
168        let id = ctx.id().to_owned();
169        let verifier = ProxyVerifier::new(prf, vm, ctx);
170
171        Self {
172            verifier: Box::new(verifier),
173            id,
174        }
175    }
176
177    pub(crate) async fn setup(&mut self) -> Result<(), Error> {
178        self.verifier.alloc()?;
179
180        debug!("setting up proxy-tls");
181        self.verifier.preprocess().await?;
182
183        Ok(())
184    }
