26pub(crate) async fn prove<T: Vm<Binary> + MacStore + Send + Sync>(
27    ctx: &mut Context,
28    vm: &mut T,
29    keys: &SessionKeys,
30    transcript: &Transcript,
31    tls_transcript: &TlsTranscript,
32    config: &ProveConfig,
33) -> Result<ProverOutput, ProverError> {
34    let mut output = ProverOutput {
35        transcript_commitments: Vec::default(),
36        transcript_secrets: Vec::default(),
37    };
38
39    let (reveal_sent, reveal_recv) = config.reveal().cloned().unwrap_or_default();
40    let (mut commit_sent, mut commit_recv) = (RangeSet::default(), RangeSet::default());
41    if let Some(commit_config) = config.transcript_commit() {
42        commit_config
43            .iter_hash()
44            .for_each(|((direction, idx), _)| match direction {
45                Direction::Sent => commit_sent.union_mut(idx),
46                Direction::Received => commit_recv.union_mut(idx),
47            });
48
49        commit_config
50            .iter_encoding()
51            .for_each(|(direction, idx)| match direction {
52                Direction::Sent => commit_sent.union_mut(idx),
53                Direction::Received => commit_recv.union_mut(idx),
54            });
55    }
56
57    let transcript_refs = TranscriptRefs {
58        sent: prove_plaintext(
59            vm,
60            keys.client_write_key,
61            keys.client_write_iv,
62            transcript.sent(),
63            tls_transcript
64                .sent()
65                .iter()
66                .filter(|record| record.typ == ContentType::ApplicationData),
67            &reveal_sent,
68            &commit_sent,
69        )
70        .map_err(ProverError::commit)?,
71        recv: prove_plaintext(
72            vm,
73            keys.server_write_key,
74            keys.server_write_iv,
75            transcript.received(),
76            tls_transcript
77                .recv()
78                .iter()
79                .filter(|record| record.typ == ContentType::ApplicationData),
80            &reveal_recv,
81            &commit_recv,
82        )
83        .map_err(ProverError::commit)?,
84    };
85
86    let hash_commitments = if let Some(commit_config) = config.transcript_commit()
87        && commit_config.has_hash()
88    {
89        Some(
90            prove_hash(
91                vm,
92                &transcript_refs,
93                commit_config
94                    .iter_hash()
95                    .map(|((dir, idx), alg)| (*dir, idx.clone(), *alg)),
96            )
97            .map_err(ProverError::commit)?,
98        )
99    } else {
100        None
101    };
102
103    vm.execute_all(ctx).await.map_err(ProverError::zk)?;
104
105    if let Some(commit_config) = config.transcript_commit()
106        && commit_config.has_encoding()
107    {
108        let mut sent_ranges = RangeSet::default();
109        let mut recv_ranges = RangeSet::default();
110        for (dir, idx) in commit_config.iter_encoding() {
111            match dir {
112                Direction::Sent => sent_ranges.union_mut(idx),
113                Direction::Received => recv_ranges.union_mut(idx),
114            }
115        }
116
117        let sent_map = transcript_refs
118            .sent
119            .index(&sent_ranges)
120            .expect("indices are valid");
121        let recv_map = transcript_refs
122            .recv
123            .index(&recv_ranges)
124            .expect("indices are valid");
125
126        let (commitment, tree) = encoding::receive(
127            ctx,
128            vm,
129            *commit_config.encoding_hash_alg(),
130            &sent_map,
131            &recv_map,
132            commit_config.iter_encoding(),
133        )
134        .await?;
135
136        output
137            .transcript_commitments
138            .push(TranscriptCommitment::Encoding(commitment));
139        output
140            .transcript_secrets
141            .push(TranscriptSecret::Encoding(tree));
142    }
143
144    if let Some((hash_fut, hash_secrets)) = hash_commitments {
145        let hash_commitments = hash_fut.try_recv().map_err(ProverError::commit)?;
146        for (commitment, secret) in hash_commitments.into_iter().zip(hash_secrets) {
147            output
148                .transcript_commitments
149                .push(TranscriptCommitment::Hash(commitment));
150            output
151                .transcript_secrets
152                .push(TranscriptSecret::Hash(secret));
153        }
154    }
155
156    Ok(output)
157}
