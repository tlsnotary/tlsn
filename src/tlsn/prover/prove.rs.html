22pub(crate) async fn prove<T: Vm<Binary> + MacStore + Send + Sync>(
23    ctx: &mut Context,
24    vm: &mut T,
25    keys: &SessionKeys,
26    server_name: &ServerName,
27    transcript: &Transcript,
28    tls_transcript: &TlsTranscript,
29    config: &ProveConfig,
30) -> Result<ProverOutput, ProverError> {
31    let mut output = ProverOutput {
32        transcript_commitments: Vec::default(),
33        transcript_secrets: Vec::default(),
34    };
35
36    let request = ProveRequest {
37        handshake: config.server_identity().then(|| {
38            (
39                server_name.clone(),
40                HandshakeData {
41                    certs: tls_transcript
42                        .server_cert_chain()
43                        .expect("server cert chain is present")
44                        .to_vec(),
45                    sig: tls_transcript
46                        .server_signature()
47                        .expect("server signature is present")
48                        .clone(),
49                    binding: tls_transcript.certificate_binding().clone(),
50                },
51            )
52        }),
53        transcript: config
54            .reveal()
55            .map(|(sent, recv)| transcript.to_partial(sent.clone(), recv.clone())),
56        transcript_commit: config.transcript_commit().map(|config| config.to_request()),
57    };
58
59    ctx.io_mut()
60        .send(request)
61        .await
62        .map_err(ProverError::from)?;
63
64    let mut auth_sent_ranges = RangeSet::default();
65    let mut auth_recv_ranges = RangeSet::default();
66
67    let (reveal_sent, reveal_recv) = config.reveal().cloned().unwrap_or_default();
68
69    auth_sent_ranges.union_mut(&reveal_sent);
70    auth_recv_ranges.union_mut(&reveal_recv);
71
72    if let Some(commit_config) = config.transcript_commit() {
73        commit_config
74            .iter_hash()
75            .for_each(|((direction, idx), _)| match direction {
76                Direction::Sent => auth_sent_ranges.union_mut(idx),
77                Direction::Received => auth_recv_ranges.union_mut(idx),
78            });
79
80        commit_config
81            .iter_encoding()
82            .for_each(|(direction, idx)| match direction {
83                Direction::Sent => auth_sent_ranges.union_mut(idx),
84                Direction::Received => auth_recv_ranges.union_mut(idx),
85            });
86    }
87
88    let mut zk_aes_sent = ZkAesCtr::new(
89        keys.client_write_key,
90        keys.client_write_iv,
91        tls_transcript
92            .sent()
93            .iter()
94            .filter(|record| record.typ == ContentType::ApplicationData),
95    );
96    let mut zk_aes_recv = ZkAesCtr::new(
97        keys.server_write_key,
98        keys.server_write_iv,
99        tls_transcript
100            .recv()
101            .iter()
102            .filter(|record| record.typ == ContentType::ApplicationData),
103    );
104
105    let sent_refs = prove_plaintext(
106        vm,
107        &mut zk_aes_sent,
108        transcript.sent(),
109        &auth_sent_ranges,
110        &reveal_sent,
111    )
112    .map_err(ProverError::commit)?;
113    let recv_refs = prove_plaintext(
114        vm,
115        &mut zk_aes_recv,
116        transcript.received(),
117        &auth_recv_ranges,
118        &reveal_recv,
119    )
120    .map_err(ProverError::commit)?;
121
122    let transcript_refs = TranscriptRefs {
123        sent: sent_refs,
124        recv: recv_refs,
125    };
126
127    let hash_commitments = if let Some(commit_config) = config.transcript_commit()
128        && commit_config.has_hash()
129    {
130        Some(
131            prove_hash(
132                vm,
133                &transcript_refs,
134                commit_config
135                    .iter_hash()
136                    .map(|((dir, idx), alg)| (*dir, idx.clone(), *alg)),
137            )
138            .map_err(ProverError::commit)?,
139        )
140    } else {
141        None
142    };
143
144    vm.execute_all(ctx).await.map_err(ProverError::zk)?;
145
146    if let Some(commit_config) = config.transcript_commit()
147        && commit_config.has_encoding()
148    {
149        let mut sent_ranges = RangeSet::default();
150        let mut recv_ranges = RangeSet::default();
151        for (dir, idx) in commit_config.iter_encoding() {
152            match dir {
153                Direction::Sent => sent_ranges.union_mut(idx),
154                Direction::Received => recv_ranges.union_mut(idx),
155            }
156        }
157
158        let sent_map = transcript_refs
159            .sent
160            .index(&sent_ranges)
161            .expect("indices are valid");
162        let recv_map = transcript_refs
163            .recv
164            .index(&recv_ranges)
165            .expect("indices are valid");
166
167        let (commitment, tree) = encoding::receive(
168            ctx,
169            vm,
170            *commit_config.encoding_hash_alg(),
171            &sent_map,
172            &recv_map,
173            commit_config.iter_encoding(),
174        )
175        .await?;
176
177        output
178            .transcript_commitments
179            .push(TranscriptCommitment::Encoding(commitment));
180        output
181            .transcript_secrets
182            .push(TranscriptSecret::Encoding(tree));
183    }
184
185    if let Some((hash_fut, hash_secrets)) = hash_commitments {
186        let hash_commitments = hash_fut.try_recv().map_err(ProverError::commit)?;
187        for (commitment, secret) in hash_commitments.into_iter().zip(hash_secrets) {
188            output
189                .transcript_commitments
190                .push(TranscriptCommitment::Hash(commitment));
191            output
192                .transcript_secrets
193                .push(TranscriptSecret::Hash(secret));
194        }
195    }
196
197    Ok(output)
198}
