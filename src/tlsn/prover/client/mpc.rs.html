28pub(crate) struct MpcTlsClient {
29    state: State,
30    decrypt: Arc<DecryptState>,
31    client_wants_close: bool,
32    server_closed: bool,
33}
34
35enum State {
36    Start {
37        inner: Box<InnerState>,
38    },
39    Active {
40        inner: Box<InnerState>,
41    },
42    Busy {
43        fut: Pin<Box<dyn Future<Output = Result<Box<InnerState>, TlsnError>> + Send>>,
44    },
45    CloseActive {
46        inner: Box<InnerState>,
47    },
48    CloseBusy {
49        fut: Pin<Box<dyn Future<Output = Result<Box<InnerState>, TlsnError>> + Send>>,
50    },
51    Finalizing {
52        fut: Pin<FinalizeFuture>,
53    },
54    Finished,
55    Error,
56}

58impl MpcTlsClient {
59    pub(crate) fn new(
60        keys: SessionKeys,
61        vm: Arc<Mutex<Deap<ProverMpc, ProverZk>>>,
62        span: Span,
63        config: &TlsClientConfig,
64        server_name: ServerName,
65        mpc_tls: Box<MpcTlsLeader>,
66    ) -> Result<Self, TlsnError> {
67        let config = create_client_config(config)?;
68        let decrypt = DecryptState {
69            decrypt: AtomicBool::new(mpc_tls.is_decrypting()),
70        };
71
72        let tls = ClientConnection::new(Arc::new(config), mpc_tls, server_name).map_err(|e| {
73            TlsnError::config()
74                .with_msg("failed to create tls client connection")
75                .with_source(e)
76        })?;
77
78        let inner = InnerState {
79            span,
80            tls,
81            vm,
82            keys,
83            decrypt: decrypt.is_decrypting(),
84            client_closed: false,
85        };
86
87        let client = Self {
88            decrypt: Arc::new(decrypt),
89            client_wants_close: false,
90            server_closed: false,
91            state: State::Start {
92                inner: Box::new(inner),
93            },
94        };
95
96        Ok(client)
97    }
98
99    fn inner_client_mut(&mut self) -> Option<&mut ClientConnection> {
100        if let State::Active { inner } | State::CloseActive { inner } = &mut self.state {
101            Some(&mut inner.tls)
102        } else {
103            None
104        }
105    }
106
107    fn inner_client(&self) -> Option<&ClientConnection> {
108        if let State::Active { inner } | State::CloseActive { inner } = &self.state {
109            Some(&inner.tls)
110        } else {
111            None
112        }
113    }
114}

116impl TlsClient for MpcTlsClient {
117    type Error = TlsnError;
118
119    fn wants_read_tls(&self) -> bool {
120        if let Some(client) = self.inner_client() {
121            client.wants_read()
122        } else {
123            false
124        }
125    }
126
127    fn wants_write_tls(&self) -> bool {
128        if let Some(client) = self.inner_client() {
129            client.wants_write()
130        } else {
131            false
132        }
133    }
134
135    fn read_tls(&mut self, mut buf: &[u8]) -> Result<usize, Self::Error> {
136        if let Some(client) = self.inner_client_mut()
137            && client.wants_read()
138        {
139            client.read_tls(&mut buf).map_err(TlsnError::from)
140        } else {
141            Ok(0)
142        }
143    }
144
145    fn write_tls(&mut self, mut buf: &mut [u8]) -> Result<usize, Self::Error> {
146        if let Some(client) = self.inner_client_mut()
147            && client.wants_write()
148        {
149            client.write_tls(&mut buf).map_err(TlsnError::from)
150        } else {
151            Ok(0)
152        }
153    }
154
155    fn wants_read(&self) -> bool {
156        if let Some(client) = self.inner_client() {
157            !client.plaintext_is_empty()
158        } else {
159            false
160        }
161    }
162
163    fn wants_write(&self) -> bool {
164        if let Some(client) = self.inner_client() {
165            !client.sendable_plaintext_is_full()
166        } else {
167            false
168        }
169    }
170
171    fn read(&mut self, buf: &mut [u8]) -> Result<usize, Self::Error> {
172        if let Some(client) = self.inner_client_mut()
173            && !client.plaintext_is_empty()
174        {
175            client.read_plaintext(buf).map_err(TlsnError::from)
176        } else {
177            Ok(0)
178        }
179    }
180
181    fn write(&mut self, buf: &[u8]) -> Result<usize, Self::Error> {
182        if let Some(client) = self.inner_client_mut()
183            && !client.sendable_plaintext_is_full()
184        {
185            client
186                .write_plaintext(buf)
187                .map_err(|err| TlsnError::internal().with_source(err))
188        } else {
189            Ok(0)
190        }
191    }
192
193    fn client_close(&mut self) {
194        self.client_wants_close = true;
195    }
196
197    fn server_close(&mut self) {
198        self.server_closed = true;
199    }

201    fn decrypt(&self) -> Arc<DecryptState> {
202        self.decrypt.clone()
203    }
204
205    fn poll(
206        &mut self,
207        cx: &mut std::task::Context,
208    ) -> Poll<Result<(Context, ProverZk, TlsOutput), Self::Error>> {
209        match std::mem::replace(&mut self.state, State::Error) {
210            State::Start { inner } => {
211                trace!("inner client is starting");
212                self.state = State::Busy {
213                    fut: Box::pin(inner.start()),
214                };
215                self.poll(cx)
216            }
217            State::Active { mut inner } => {
218                trace!("inner client is active");
219                let decrypt = self.decrypt.is_decrypting();
220
221                if !inner.tls.is_handshaking() {
222                    if self.server_closed {
223                        self.state = State::CloseBusy {
224                            fut: Box::pin(inner.server_close()),
225                        };
226                    } else if self.client_wants_close {
227                        self.state = State::Busy {
228                            fut: Box::pin(inner.client_close()),
229                        };
230                    } else if decrypt != inner.decrypt {
231                        inner
232                            .tls
233                            .backend_mut()
234                            .enable_decryption(decrypt)
235                            .map_err(|err| TlsnError::internal().with_source(err))?;
236
237                        inner.decrypt = decrypt;
238                        self.state = State::Busy {
239                            fut: Box::pin(inner.run()),
240                        };
241                    } else {
242                        self.state = State::Busy {
243                            fut: Box::pin(inner.run()),
244                        };
245                    }
246                    return self.poll(cx);
247                }
248                self.state = State::Busy {
249                    fut: Box::pin(inner.run()),
250                };
251                self.poll(cx)
252            }
253            State::Busy { mut fut } => {
254                trace!("inner client is busy");
255
256                match fut.as_mut().poll(cx)? {
257                    Poll::Ready(inner) => {
258                        self.state = State::Active { inner };
259                        // re-poll immediately if a close transition is pending
260                        if self.server_closed {
261                            return self.poll(cx);
262                        }
263                    }
264                    Poll::Pending => self.state = State::Busy { fut },
265                }
266                Poll::Pending
267            }
268            State::CloseActive { inner } => {
269                trace!("inner client is active closing");
270                self.state = State::CloseBusy {
271                    fut: Box::pin(inner.run()),
272                };
273                self.poll(cx)
274            }
275            State::CloseBusy { mut fut } => {
276                trace!("inner client is busy closing");
277
278                match fut.as_mut().poll(cx)? {
279                    Poll::Ready(mut inner) if inner.is_record_layer_empty() => {
280                        let (ctx, transcript) = inner
281                            .tls
282                            .backend_mut()
283                            .finish()
284                            .expect("connection should be closed");
285                        self.state = State::Finalizing {
286                            fut: Box::pin(inner.finalize(ctx, transcript)),
287                        };
288                        self.poll(cx)
289                    }
290                    Poll::Ready(inner) => {
291                        cx.waker().wake_by_ref();
292                        self.state = State::CloseActive { inner };
293                        Poll::Pending
294                    }
295                    Poll::Pending => {
296                        self.state = State::CloseBusy { fut };
297                        Poll::Pending
298                    }
299                }
300            }
301            State::Finalizing { mut fut } => match fut.poll_unpin(cx) {
302                Poll::Ready(output) => {
303                    let (inner, ctx, tls_transcript) = output?;
304                    let InnerState { vm, keys, .. } = inner;
305
306                    let (_, vm) = Arc::into_inner(vm)
307                        .expect("vm should have only 1 reference")
308                        .into_inner()
309                        .into_inner();
310
311                    let output = TlsOutput {
312                        keys,
313                        tls_transcript,
314                    };
315
316                    self.state = State::Finished;
317                    Poll::Ready(Ok((ctx, vm, output)))
318                }
319                Poll::Pending => {
320                    self.state = State::Finalizing { fut };
321                    Poll::Pending
322                }
323            },
324            State::Finished => Poll::Ready(Err(
325                TlsnError::internal().with_msg("mpc tls client polled again in finished state")
326            )),
327            State::Error => Poll::Ready(Err(
328                TlsnError::internal().with_msg("mpc tls client is in error state")
329            )),
330        }
331    }
332}

334struct InnerState {
335    span: Span,
336    tls: ClientConnection,
337    vm: Arc<Mutex<Deap<ProverMpc, ProverZk>>>,
338    keys: SessionKeys,
339    decrypt: bool,
340    client_closed: bool,
341}
342
343impl InnerState {
344    #[instrument(parent = &self.span, level = "debug", skip_all, err)]
345    async fn start(mut self: Box<Self>) -> Result<Box<Self>, TlsnError> {
346        self.tls
347            .start()
348            .await
349            .map_err(|err| TlsnError::internal().with_source(err))?;
350        Ok(self)
351    }
352
353    #[instrument(parent = &self.span, level = "trace", skip_all, err)]
354    async fn run(mut self: Box<Self>) -> Result<Box<Self>, TlsnError> {
355        let mut state = self
356            .tls
357            .process_new_packets()
358            .await
359            .map_err(|err| TlsnError::internal().with_source(err))?;
360        loop {
361            let new_state = self
362                .tls
363                .process_new_packets()
364                .await
365                .map_err(|err| TlsnError::internal().with_source(err))?;
366
367            if new_state.plaintext_bytes_to_read() == state.plaintext_bytes_to_read()
368                && new_state.tls_bytes_to_write() == state.tls_bytes_to_write()
369            {
370                break;
371            }
372            state = new_state;
373        }
374
375        Ok(self)
376    }
377
378    #[instrument(parent = &self.span, level = "debug", skip_all, err)]
379    async fn client_close(mut self: Box<Self>) -> Result<Box<Self>, TlsnError> {
380        self = self.run().await?;
381
382        if !self.client_closed {
383            debug!("sending close notify");
384            if let Err(e) = self.tls.send_close_notify().await {
385                warn!("failed to send close_notify to server: {}", e);
386            }
387            self.client_closed = true;
388        }
389
390        Ok(self)
391    }
392
393    #[instrument(parent = &self.span, level = "debug", skip_all, err)]
394    async fn server_close(mut self: Box<Self>) -> Result<Box<Self>, TlsnError> {
395        self = self.run().await?;
396
397        self.tls
398            .server_closed()
399            .await
400            .map_err(|err| TlsnError::internal().with_source(err))?;
401        debug!("closed connection serverside");
402
403        Ok(self)
404    }
405
406    #[instrument(parent = &self.span, level = "debug", skip_all, err)]
407    async fn finalize(
408        self,
409        mut ctx: Context,
410        transcript: TlsTranscript,
411    ) -> Result<(Self, Context, TlsTranscript), TlsnError> {
412        {
413            let mut vm = self.vm.try_lock().expect("VM should not be locked");
414
415            // Finalize DEAP.
416            vm.finalize(&mut ctx)
417                .await
418                .map_err(|err| TlsnError::internal().with_source(err))?;
419
420            debug!("mpc finalized");
421        }
422
423        debug!("MPC-TLS done");
424        Ok((self, ctx, transcript))
425    }
426
427    fn is_record_layer_empty(&self) -> bool {
428        self.tls.plaintext_is_empty() && self.tls.is_empty().unwrap_or(false)
429    }
430}

432fn create_client_config(config: &TlsClientConfig) -> Result<tls_client::ClientConfig, TlsnError> {
433    let root_store = tls_client::RootCertStore {
434        roots: config
435            .root_store()
436            .roots
437            .iter()
438            .map(|cert| {
439                let der = CertificateDer::from_slice(&cert.0);
440                anchor_from_trusted_cert(&der)
441                    .map(|anchor| anchor.to_owned())
442                    .map_err(|e| {
443                        TlsnError::config()
444                            .with_msg("failed to parse root certificate")
445                            .with_source(e)
446                    })
447            })
448            .collect::<Result<Vec<_>, _>>()?,
449    };
450
451    let rustls_config = tls_client::ClientConfig::builder()
452        .with_safe_defaults()
453        .with_root_certificates(root_store);
454
455    let rustls_config = if let Some((cert, key)) = config.client_auth() {
456        rustls_config
457            .with_single_cert(
458                cert.iter()
459                    .map(|cert| tls_client::Certificate(cert.0.clone()))
460                    .collect(),
461                tls_client::PrivateKey(key.0.clone()),
462            )
463            .map_err(|e| {
464                TlsnError::config()
465                    .with_msg("failed to configure client authentication")
466                    .with_source(e)
467            })?
468    } else {
469        rustls_config.with_no_client_auth()
470    };
471
472    Ok(rustls_config)
