44    pub(crate) fn build_mpc_tls_config(&self) -> Config {
45        let mut builder = Config::builder();
46
47        builder
48            .defer_decryption(self.protocol_config.defer_decryption_from_start())
49            .max_sent(self.protocol_config.max_sent_data())
50            .max_recv_online(self.protocol_config.max_recv_data_online())
51            .max_recv(self.protocol_config.max_recv_data());
52
53        if let Some(max_sent_records) = self.protocol_config.max_sent_records() {
54            builder.max_sent_records(max_sent_records);
55        }
56
57        if let Some(max_recv_records_online) = self.protocol_config.max_recv_records_online() {
58            builder.max_recv_records_online(max_recv_records_online);
59        }
60
61        if let NetworkSetting::Latency = self.protocol_config.network() {
62            builder.low_bandwidth();
63        }
64
65        builder.build().unwrap()
66    }

69/// Configuration for the prover's TLS connection.

71pub struct TlsConfig {
72    /// Root certificates.
73    root_store: RootCertStore,
74    /// Certificate chain and a matching private key for client
75    /// authentication.
76    client_auth: Option<(Vec<key::Certificate>, key::PrivateKey)>,
77}
78
79impl Default for TlsConfig {
80    fn default() -> Self {
81        let mut root_store = RootCertStore::empty();
82        root_store.add_server_trust_anchors(webpki_roots::TLS_SERVER_ROOTS.iter().map(|ta| {
83            OwnedTrustAnchor::from_subject_spki_name_constraints(
84                ta.subject.as_ref(),
85                ta.subject_public_key_info.as_ref(),
86                ta.name_constraints.as_ref().map(|nc| nc.as_ref()),
87            )
88        }));
89
90        Self {
91            root_store,
92            client_auth: None,
93        }
94    }
95}

114/// Builder for [`TlsConfig`].
115#[derive(Debug, Default)]
116pub struct TlsConfigBuilder {
117    root_store: Option<RootCertStore>,
118    client_auth: Option<(Vec<key::Certificate>, key::PrivateKey)>,
119}
120
121impl TlsConfigBuilder {
122    /// Sets the root certificates to use for verifying the server's
123    /// certificate.
124    pub fn root_store(&mut self, store: RootCertStore) -> &mut Self {
125        self.root_store = Some(store);
126        self
127    }
128
129    /// Sets a DER-encoded certificate chain and a matching private key for
130    /// client authentication.
131    ///
132    /// Often the chain will consist of a single end-entity certificate.
133    ///
134    /// # Arguments
135    ///
136    /// * `cert_key` - A tuple containing the certificate chain and the private
137    ///   key.
138    ///
139    ///   - Each certificate in the chain must be in the X.509 format.
140    ///   - The key must be in the ASN.1 format (either PKCS#8 or PKCS#1).
141    pub fn client_auth(&mut self, cert_key: (Vec<Vec<u8>>, Vec<u8>)) -> &mut Self {
142        let certs = cert_key
143            .0
144            .into_iter()
145            .map(key::Certificate)
146            .collect::<Vec<_>>();
147
148        self.client_auth = Some((certs, key::PrivateKey(cert_key.1)));
149        self
150    }
151
152    /// Sets a PEM-encoded certificate chain and a matching private key for
153    /// client authentication.
154    ///
155    /// Often the chain will consist of a single end-entity certificate.
156    ///
157    /// # Arguments
158    ///
159    /// * `cert_key` - A tuple containing the certificate chain and the private
160    ///   key.
161    ///
162    ///   - Each certificate in the chain must be in the X.509 format.
163    ///   - The key must be in the ASN.1 format (either PKCS#8 or PKCS#1).
164    pub fn client_auth_pem(
165        &mut self,
166        cert_key: (Vec<Vec<u8>>, Vec<u8>),
167    ) -> Result<&mut Self, TlsConfigError> {
168        let key = match PrivatePkcs8KeyDer::from_pem_slice(&cert_key.1) {
169            // Try to parse as PEM PKCS#8.
170            Ok(key) => (*key.secret_pkcs8_der()).to_vec(),
171            // Otherwise, try to parse as PEM PKCS#1.
172            Err(_) => match PrivatePkcs1KeyDer::from_pem_slice(&cert_key.1) {
173                Ok(key) => (*key.secret_pkcs1_der()).to_vec(),
174                Err(_) => return Err(ErrorRepr::InvalidKey.into()),
175            },
176        };
177
178        let certs = cert_key
179            .0
180            .iter()
181            .map(|c| {
182                let c =
183                    CertificateDer::from_pem_slice(c).map_err(|_| ErrorRepr::InvalidCertificate)?;
184                Ok::<key::Certificate, TlsConfigError>(key::Certificate(c.as_ref().to_vec()))
185            })
186            .collect::<Result<Vec<_>, _>>()?;
187
188        self.client_auth = Some((certs, key::PrivateKey(key)));
189        Ok(self)
190    }
191
192    /// Builds the TLS configuration.
193    pub fn build(&self) -> Result<TlsConfig, TlsConfigError> {
194        Ok(TlsConfig {
195            root_store: self.root_store.clone().unwrap_or_else(|| {
196                let mut root_store = RootCertStore::empty();
197                root_store.add_server_trust_anchors(webpki_roots::TLS_SERVER_ROOTS.iter().map(
198                    |ta| {
199                        OwnedTrustAnchor::from_subject_spki_name_constraints(
200                            ta.subject.as_ref(),
201                            ta.subject_public_key_info.as_ref(),
202                            ta.name_constraints.as_ref().map(|nc| nc.as_ref()),
203                        )
204                    },
205                ));
206                root_store
207            }),
208            client_auth: self.client_auth.clone(),
209        })
210    }
211}
