19/// TLS client for MPC and proxy-based TLS implementations.
20pub(crate) trait TlsClient {
21    type Error: std::error::Error + Send + Sync + Unpin + 'static;
22
23    /// Returns `true` if the client wants to read TLS data from the server.
24    fn wants_read_tls(&self) -> bool;
25
26    /// Returns `true` if the client wants to write TLS data to the server.
27    fn wants_write_tls(&self) -> bool;
28
29    /// Reads TLS data from the server.
30    fn read_tls(&mut self, buf: &[u8]) -> Result<usize, Self::Error>;
31
32    /// Writes TLS data for the server into the provided buffer.
33    fn write_tls(&mut self, buf: &mut [u8]) -> Result<usize, Self::Error>;
34
35    /// Returns `true` if the client wants to read plaintext data.
36    fn wants_read(&self) -> bool;
37
38    /// Returns `true` if the client wants to write plaintext data.
39    fn wants_write(&self) -> bool;
40
41    /// Reads plaintext data from the server into the provided buffer.
42    fn read(&mut self, buf: &mut [u8]) -> Result<usize, Self::Error>;
43
44    /// Writes plaintext data to be sent to the server.
45    fn write(&mut self, buf: &[u8]) -> Result<usize, Self::Error>;
46
47    /// Client closes the connection.
48    fn client_close(&mut self);
49
50    /// Server closes the connection.
51    fn server_close(&mut self);
52
53    /// Returns the decryption state.
54    fn decrypt(&self) -> Arc<DecryptState>;
55
56    /// Polls the client to make progress.
57    fn poll(
58        &mut self,
59        cx: &mut std::task::Context,
60    ) -> Poll<Result<(Context, ProverZk, TlsOutput), Self::Error>>;
61}
62
63/// Decryption state.
64#[derive(Debug)]
65pub(crate) struct DecryptState {
66    decrypt: AtomicBool,
67}
68
69impl DecryptState {
70    pub(crate) fn enable_decryption(&self, enable: bool) {
71        self.decrypt.store(enable, Ordering::Release);
72    }
73
74    pub(crate) fn is_decrypting(&self) -> bool {
75        self.decrypt.load(Ordering::Acquire)
76    }
77}
