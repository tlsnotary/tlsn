9/// References to the application plaintext in the transcript.
10#[derive(Debug, Default, Clone)]
11pub(crate) struct TranscriptRefs {
12    sent: Vec<Vector<U8>>,
13    recv: Vec<Vector<U8>>,
14}
15
16impl TranscriptRefs {
17    pub(crate) fn new(sent: Vec<Vector<U8>>, recv: Vec<Vector<U8>>) -> Self {
18        Self { sent, recv }
19    }
20
21    /// Returns the sent plaintext references.
22    pub(crate) fn sent(&self) -> &[Vector<U8>] {
23        &self.sent
24    }
25
26    /// Returns the received plaintext references.
27    pub(crate) fn recv(&self) -> &[Vector<U8>] {
28        &self.recv
29    }
30
31    /// Returns the transcript lengths.
32    pub(crate) fn len(&self) -> (usize, usize) {
33        let sent = self.sent.iter().map(|v| v.len()).sum();
34        let recv = self.recv.iter().map(|v| v.len()).sum();
35
36        (sent, recv)
37    }
38
39    /// Returns VM references for the given direction and index, otherwise
40    /// `None` if the index is out of bounds.
41    pub(crate) fn get(&self, direction: Direction, idx: &Idx) -> Option<Vec<Vector<U8>>> {
42        if idx.is_empty() {
43            return Some(Vec::new());
44        }
45
46        let refs = match direction {
47            Direction::Sent => &self.sent,
48            Direction::Received => &self.recv,
49        };
50
51        // Computes the transcript range for each reference.
52        let mut start = 0;
53        let mut slice_iter = refs.iter().map(move |slice| {
54            let out = (slice, start..start + slice.len());
55            start += slice.len();
56            out
57        });
58
59        let mut slices = Vec::new();
60        let (mut slice, mut slice_range) = slice_iter.next()?;
61        for range in idx.iter_ranges() {
62            loop {
63                if let Some(intersection) = slice_range.intersection(&range) {
64                    let start = intersection.start - slice_range.start;
65                    let end = intersection.end - slice_range.start;
66                    slices.push(slice.get(start..end).expect("range should be in bounds"));
67                }
68
69                // Proceed to next range if the current slice extends beyond. Otherwise, proceed
70                // to the next slice.
71                if range.end <= slice_range.end {
72                    break;
73                } else {
74                    (slice, slice_range) = slice_iter.next()?;
75                }
76            }
77        }
78
79        Some(slices)
80    }
81}

83/// Decodes the transcript.
84pub(crate) fn decode_transcript(
85    vm: &mut dyn Vm<Binary>,
86    sent: &Idx,
87    recv: &Idx,
88    refs: &TranscriptRefs,
89) -> Result<(), VmError> {
90    let sent_refs = refs.get(Direction::Sent, sent).expect("index is in bounds");
91    let recv_refs = refs
92        .get(Direction::Received, recv)
93        .expect("index is in bounds");
94
95    for slice in sent_refs.into_iter().chain(recv_refs) {
96        // Drop the future, we don't need it.
97        drop(vm.decode(slice)?);
98    }
99
100    Ok(())
101}

103/// Verifies a partial transcript.
104pub(crate) fn verify_transcript(
105    vm: &mut dyn Vm<Binary>,
106    transcript: &PartialTranscript,
107    refs: &TranscriptRefs,
108) -> Result<(), InconsistentTranscript> {
109    let sent_refs = refs
110        .get(Direction::Sent, transcript.sent_authed())
111        .expect("index is in bounds");
112    let recv_refs = refs
113        .get(Direction::Received, transcript.received_authed())
114        .expect("index is in bounds");
115
116    let mut authenticated_data = Vec::new();
117    for data in sent_refs.into_iter().chain(recv_refs) {
118        let plaintext = vm
119            .get(data)
120            .expect("reference is valid")
121            .expect("plaintext is decoded");
122        authenticated_data.extend_from_slice(&plaintext);
123    }
124
125    let mut purported_data = Vec::with_capacity(authenticated_data.len());
126    for range in transcript.sent_authed().iter_ranges() {
127        purported_data.extend_from_slice(&transcript.sent_unsafe()[range]);
128    }
129
130    for range in transcript.received_authed().iter_ranges() {
131        purported_data.extend_from_slice(&transcript.received_unsafe()[range]);
132    }
133
134    if purported_data != authenticated_data {
135        return Err(InconsistentTranscript {});
136    }
137
138    Ok(())
139}

154    // TRANSCRIPT_REFS:

156    // 48..96 -> 6 slots
157    // 112..176 -> 8 slots
158    // 240..288 -> 6 slots
159    // 352..392 -> 5 slots
160    // 440..480 -> 5 slots

162
163    const IDXS: &[Range<usize>] = &[0..4, 5..10, 14..16, 16..28];
164
165    // 1. Take slots 0..4,   4  slots -> 48..80 (4)
166    // 2. Take slots 5..10,  5  slots -> 88..96 (1) + 112..144 (4)
167    // 3. Take slots 14..16, 2  slots -> 240..256 (2)
168    // 4. Take slots 16..28, 12 slots -> 256..288 (4) + 352..392 (5) + 440..464 (3)
169    //
170    // 5. Merge slots 240..256 and 256..288 => 240..288 and get EXPECTED_REFS
171    const EXPECTED_REFS: &[Range<usize>] =
172        &[48..80, 88..96, 112..144, 240..288, 352..392, 440..464];
173
174    #[test]
175    fn test_transcript_refs_get() {
176        let transcript_refs: Vec<Vector<U8>> = TRANSCRIPT_REFS
177            .iter()
178            .cloned()
179            .map(|range| Vector::from_raw(Slice::from_range_unchecked(range)))
180            .collect();
181
182        let transcript_refs = TranscriptRefs {
183            sent: transcript_refs.clone(),
184            recv: transcript_refs,
185        };
186
187        let vm_refs = transcript_refs
188            .get(Direction::Sent, &idx_fixture())
189            .unwrap();
190
191        let expected_refs: Vec<Vector<U8>> = EXPECTED_REFS
192            .iter()
193            .cloned()
194            .map(|range| Vector::from_raw(Slice::from_range_unchecked(range)))
195            .collect();
196
197        assert_eq!(
198            vm_refs.len(),
199            expected_refs.len(),
200            "Length of actual and expected refs are not equal"
201        );
202
203        for (&expected, actual) in expected_refs.iter().zip(vm_refs) {
204            assert_eq!(expected, actual);
205        }
206    }
