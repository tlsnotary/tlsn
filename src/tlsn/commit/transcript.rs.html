9/// References to the application plaintext in the transcript.
10#[derive(Debug, Default, Clone)]
11pub(crate) struct TranscriptRefs {
12    sent: Vec<Vector<U8>>,
13    recv: Vec<Vector<U8>>,
14}
15
16impl TranscriptRefs {
17    pub(crate) fn new(sent: Vec<Vector<U8>>, recv: Vec<Vector<U8>>) -> Self {
18        Self { sent, recv }
19    }
20
21    /// Returns the sent plaintext references.
22    pub(crate) fn sent(&self) -> &[Vector<U8>] {
23        &self.sent
24    }
25
26    /// Returns the received plaintext references.
27    pub(crate) fn recv(&self) -> &[Vector<U8>] {
28        &self.recv
29    }
30
31    /// Returns VM references for the given direction and index, otherwise
32    /// `None` if the index is out of bounds.
33    pub(crate) fn get(&self, direction: Direction, idx: &Idx) -> Option<Vec<Vector<U8>>> {
34        if idx.is_empty() {
35            return Some(Vec::new());
36        }
37
38        let refs = match direction {
39            Direction::Sent => &self.sent,
40            Direction::Received => &self.recv,
41        };
42
43        // Computes the transcript range for each reference.
44        let mut start = 0;
45        let mut slice_iter = refs.iter().map(move |slice| {
46            let out = (slice, start..start + slice.len());
47            start += slice.len();
48            out
49        });
50
51        let mut slices = Vec::new();
52        let (mut slice, mut slice_range) = slice_iter.next()?;
53        for range in idx.iter_ranges() {
54            loop {
55                if let Some(intersection) = slice_range.intersection(&range) {
56                    let start = intersection.start - slice_range.start;
57                    let end = intersection.end - slice_range.start;
58                    slices.push(slice.get(start..end).expect("range should be in bounds"));
59                }
60
61                // Proceed to next range if the current slice extends beyond. Otherwise, proceed
62                // to the next slice.
63                if range.end <= slice_range.end {
64                    break;
65                } else {
66                    (slice, slice_range) = slice_iter.next()?;
67                }
68            }
69        }
70
71        Some(slices)
72    }
73}

75/// Decodes the transcript.
76pub(crate) fn decode_transcript(
77    vm: &mut dyn Vm<Binary>,
78    sent: &Idx,
79    recv: &Idx,
80    refs: &TranscriptRefs,
81) -> Result<(), VmError> {
82    let sent_refs = refs.get(Direction::Sent, sent).expect("index is in bounds");
83    let recv_refs = refs
84        .get(Direction::Received, recv)
85        .expect("index is in bounds");
86
87    for slice in sent_refs.into_iter().chain(recv_refs) {
88        // Drop the future, we don't need it.
89        drop(vm.decode(slice)?);
90    }
91
92    Ok(())
93}

95/// Verifies a partial transcript.
96pub(crate) fn verify_transcript(
97    vm: &mut dyn Vm<Binary>,
98    transcript: &PartialTranscript,
99    refs: &TranscriptRefs,
100) -> Result<(), InconsistentTranscript> {
101    let sent_refs = refs
102        .get(Direction::Sent, transcript.sent_authed())
103        .expect("index is in bounds");
104    let recv_refs = refs
105        .get(Direction::Received, transcript.received_authed())
106        .expect("index is in bounds");
107
108    let mut authenticated_data = Vec::new();
109    for data in sent_refs.into_iter().chain(recv_refs) {
110        let plaintext = vm
111            .get(data)
112            .expect("reference is valid")
113            .expect("plaintext is decoded");
114        authenticated_data.extend_from_slice(&plaintext);
115    }
116
117    let mut purported_data = Vec::with_capacity(authenticated_data.len());
118    for range in transcript.sent_authed().iter_ranges() {
119        purported_data.extend_from_slice(&transcript.sent_unsafe()[range]);
120    }
121
122    for range in transcript.received_authed().iter_ranges() {
123        purported_data.extend_from_slice(&transcript.received_unsafe()[range]);
124    }
125
126    if purported_data != authenticated_data {
127        return Err(InconsistentTranscript {});
128    }
129
130    Ok(())
131}

155    const IDXS: &[Range<usize>] = &[0..4, 5..10, 14..16, 16..28];

157    // 1. Take slots 0..4,   4  slots -> 48..80 (4)
158    // 2. Take slots 5..10,  5  slots -> 88..96 (1) + 112..144 (4)
159    // 3. Take slots 14..16, 2  slots -> 240..256 (2)
160    // 4. Take slots 16..28, 12 slots -> 256..288 (4) + 352..392 (5) + 440..464 (3)
161    //
162    // 5. Merge slots 240..256 and 256..288 => 240..288 and get EXPECTED_REFS
163    const EXPECTED_REFS: &[Range<usize>] =
164        &[48..80, 88..96, 112..144, 240..288, 352..392, 440..464];
165
166    #[test]
167    fn test_transcript_refs_get() {
168        let transcript_refs: Vec<Vector<U8>> = TRANSCRIPT_REFS
169            .iter()
170            .cloned()
171            .map(|range| Vector::from_raw(Slice::from_range_unchecked(range)))
172            .collect();
173
174        let transcript_refs = TranscriptRefs {
175            sent: transcript_refs.clone(),
176            recv: transcript_refs,
177        };
178
179        let vm_refs = transcript_refs
180            .get(Direction::Sent, &idx_fixture())
181            .unwrap();
182
183        let expected_refs: Vec<Vector<U8>> = EXPECTED_REFS
184            .iter()
185            .cloned()
186            .map(|range| Vector::from_raw(Slice::from_range_unchecked(range)))
187            .collect();
188
189        assert_eq!(
190            vm_refs.len(),
191            expected_refs.len(),
192            "Length of actual and expected refs are not equal"
193        );
194
195        for (&expected, actual) in expected_refs.iter().zip(vm_refs) {
196            assert_eq!(expected, actual);
197        }
198    }
