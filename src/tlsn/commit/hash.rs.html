22/// Future which will resolve to the committed hash values.
23#[derive(Debug)]
24
25pub(crate) struct HashCommitFuture {
26    #[allow(clippy::type_complexity)]
27    futs: Vec<(
28        Direction,
29        Idx,
30        HashAlgId,
31        DecodeFutureTyped<BitVec, Vec<u8>>,
32    )>,
33}
34
35impl HashCommitFuture {
36    /// Tries to receive the value, returning an error if the value is not
37    /// ready.
38    pub(crate) fn try_recv(self) -> Result<Vec<PlaintextHash>, HashCommitError> {
39        let mut output = Vec::new();
40        for (direction, idx, alg, mut fut) in self.futs {
41            let hash = fut
42                .try_recv()
43                .map_err(|_| HashCommitError::decode())?
44                .ok_or_else(HashCommitError::decode)?;
45            output.push(PlaintextHash {
46                direction,
47                idx,
48                hash: TypedHash {
49                    alg,
50                    value: Hash::try_from(hash).map_err(HashCommitError::convert)?,
51                },
52            });
53        }
54
55        Ok(output)
56    }
57}

59/// Prove plaintext hash commitments.
60pub(crate) fn prove_hash(
61    vm: &mut dyn Vm<Binary>,
62    refs: &TranscriptRefs,
63    idxs: impl IntoIterator<Item = (Direction, Idx, HashAlgId)>,
64) -> Result<(HashCommitFuture, Vec<PlaintextHashSecret>), HashCommitError> {
65    let mut futs = Vec::new();
66    let mut secrets = Vec::new();
67    for (direction, idx, alg, hash_ref, blinder_ref) in
68        hash_commit_inner(vm, Role::Prover, refs, idxs)?
69    {
70        let blinder: Blinder = rand::random();
71
72        vm.assign(blinder_ref, blinder.as_bytes().to_vec())?;
73        vm.commit(blinder_ref)?;
74
75        let hash_fut = vm.decode(Vector::<U8>::from(hash_ref))?;
76
77        futs.push((direction, idx.clone(), alg, hash_fut));
78        secrets.push(PlaintextHashSecret {
79            direction,
80            idx,
81            blinder,
82            alg,
83        });
84    }
85
86    Ok((HashCommitFuture { futs }, secrets))
87}

89/// Verify plaintext hash commitments.
90pub(crate) fn verify_hash(
91    vm: &mut dyn Vm<Binary>,
92    refs: &TranscriptRefs,
93    idxs: impl IntoIterator<Item = (Direction, Idx, HashAlgId)>,
94) -> Result<HashCommitFuture, HashCommitError> {
95    let mut futs = Vec::new();
96    for (direction, idx, alg, hash_ref, blinder_ref) in
97        hash_commit_inner(vm, Role::Verifier, refs, idxs)?
98    {
99        vm.commit(blinder_ref)?;
100
101        let hash_fut = vm.decode(Vector::<U8>::from(hash_ref))?;
102
103        futs.push((direction, idx, alg, hash_fut));
104    }
105
106    Ok(HashCommitFuture { futs })
107}

109/// Commit plaintext hashes of the transcript.
110#[allow(clippy::type_complexity)]
111fn hash_commit_inner(
112    vm: &mut dyn Vm<Binary>,
113    role: Role,
114    refs: &TranscriptRefs,
115    idxs: impl IntoIterator<Item = (Direction, Idx, HashAlgId)>,
116) -> Result<Vec<(Direction, Idx, HashAlgId, Array<U8, 32>, Vector<U8>)>, HashCommitError> {
117    let mut output = Vec::new();
118    let mut hashers = HashMap::new();
119    for (direction, idx, alg) in idxs {
120        let blinder = vm.alloc_vec::<U8>(16)?;
121        match role {
122            Role::Prover => vm.mark_private(blinder)?,
123            Role::Verifier => vm.mark_blind(blinder)?,
124        }
125
126        let hash = match alg {
127            HashAlgId::SHA256 => {
128                let mut hasher = if let Some(hasher) = hashers.get(&alg).cloned() {
129                    hasher
130                } else {
131                    let hasher = Sha256::new_with_init(vm).map_err(HashCommitError::hasher)?;
132                    hashers.insert(alg, hasher.clone());
133                    hasher
134                };
135
136                for plaintext in refs.get(direction, &idx).expect("plaintext refs are valid") {
137                    hasher.update(&plaintext);
138                }
139                hasher.update(&blinder);
140                hasher.finalize(vm).map_err(HashCommitError::hasher)?
141            }
142            alg => {
143                return Err(HashCommitError::unsupported_alg(alg));
144            }
145        };
146
147        output.push((direction, idx, alg, hash, blinder));
148    }
149
150    Ok(output)
151}

153/// Error type for hash commitments.
154#[derive(Debug, thiserror::Error)]
155#[error(transparent)]
156pub(crate) struct HashCommitError(#[from] ErrorRepr);
157
158impl HashCommitError {
159    fn decode() -> Self {
160        Self(ErrorRepr::Decode)
161    }
162
163    fn convert(e: &'static str) -> Self {
164        Self(ErrorRepr::Convert(e))
165    }
166
167    fn hasher<E>(e: E) -> Self
168    where
169        E: Into<Box<dyn std::error::Error + Send + Sync>>,
170    {
171        Self(ErrorRepr::Hasher(e.into()))
172    }
173
174    fn unsupported_alg(alg: HashAlgId) -> Self {
175        Self(ErrorRepr::UnsupportedAlg { alg })
176    }
177}
178
179#[derive(Debug, thiserror::Error)]
180#[error("hash commit error: {0}")]
181enum ErrorRepr {
182    #[error("VM error: {0}")]
183    Vm(VmError),
184    #[error("failed to decode hash")]
185    Decode,
186    #[error("failed to convert hash: {0}")]
187    Convert(&'static str),
188    #[error("unsupported hash algorithm: {alg}")]
189    UnsupportedAlg { alg: HashAlgId },
190    #[error("hasher error: {0}")]
191    Hasher(Box<dyn std::error::Error + Send + Sync>),
192}
193
194impl From<VmError> for HashCommitError {
195    fn from(value: VmError) -> Self {
196        Self(ErrorRepr::Vm(value))
197    }
