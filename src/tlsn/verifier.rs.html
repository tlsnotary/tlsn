51    /// Creates a new verifier.

53        let span = info_span!("verifier");
54        Self {
55            config,
56            span,
57            state: state::Initialized,
58        }
59    }

61    /// Sets up the verifier.
62    ///
63    /// This performs all MPC setup.
64    ///
65    /// # Arguments
66    ///
67    /// * `socket` - The socket to the prover.
68    #[instrument(parent = &self.span, level = "info", skip_all, err)]
69    pub async fn setup<S: AsyncWrite + AsyncRead + Send + Unpin + 'static>(
70        self,
71        socket: S,
72    ) -> Result<Verifier<state::Config>, VerifierError> {
73        let (mut mux_fut, mux_ctrl) = attach_mux(socket, Role::Verifier);
74        let mut mt = build_mt_context(mux_ctrl.clone());
75        let mut ctx = mux_fut.poll_with(mt.new_context()).await?;
76
77        // Receives protocol configuration from prover to perform compatibility check.
78        let SetupRequest { config, version } =
79            mux_fut.poll_with(ctx.io_mut().expect_next()).await?;
80
81        if version != *crate::config::VERSION {
82            let msg = format!(
83                "prover version does not match with verifier: {version} != {}",
84                *crate::config::VERSION
85            );
86            mux_fut
87                .poll_with(ctx.io_mut().send(Response::err(Some(msg.clone()))))
88                .await?;
89
90            // Wait for the prover to correctly close the connection.
91            if !mux_fut.is_complete() {
92                mux_ctrl.close();
93                mux_fut.await?;
94            }
95
96            return Err(VerifierError::config(msg));
97        }
98
99        Ok(Verifier {
100            config: self.config,
101            span: self.span,
102            state: state::Config {
103                mux_ctrl,
104                mux_fut,
105                ctx,
106                config,
107            },
108        })
109    }
110}

112impl Verifier<state::Config> {
113    /// Returns the proposed protocol configuration.
114    pub fn config(&self) -> &ProtocolConfig {
115        &self.state.config
116    }
117
118    /// Accepts the proposed protocol configuration.
119    #[instrument(parent = &self.span, level = "info", skip_all, err)]
120    pub async fn accept(self) -> Result<Verifier<state::Setup>, VerifierError> {
121        let state::Config {
122            mux_ctrl,
123            mut mux_fut,
124            mut ctx,
125            config,
126        } = self.state;
127
128        mux_fut.poll_with(ctx.io_mut().send(Response::ok())).await?;
129
130        let VerifierDeps { vm, mut mpc_tls } =
131            build_verifier_deps(self.config.build_mpc_tls_config(&config), ctx);
132
133        // Allocate resources for MPC-TLS in the VM.
134        let mut keys = mpc_tls.alloc()?;
135        let vm_lock = vm.try_lock().expect("VM is not locked");
136        translate_keys(&mut keys, &vm_lock);
137        drop(vm_lock);
138
139        debug!("setting up mpc-tls");
140
141        mux_fut.poll_with(mpc_tls.preprocess()).await?;
142
143        debug!("mpc-tls setup complete");
144
145        Ok(Verifier {
146            config: self.config,
147            span: self.span,
148            state: state::Setup {
149                mux_ctrl,
150                mux_fut,
151                mpc_tls,
152                keys,
153                vm,
154            },
155        })
156    }
157
158    /// Rejects the proposed protocol configuration.
159    #[instrument(parent = &self.span, level = "info", skip_all, err)]
160    pub async fn reject(self, msg: Option<&str>) -> Result<(), VerifierError> {
161        let state::Config {
162            mux_ctrl,
163            mut mux_fut,
164            mut ctx,
165            ..
166        } = self.state;
167
168        mux_fut
169            .poll_with(ctx.io_mut().send(Response::err(msg)))
170            .await?;
171
172        // Wait for the prover to correctly close the connection.
173        if !mux_fut.is_complete() {
174            mux_ctrl.close();
175            mux_fut.await?;
176        }
177
178        Ok(())
179    }
180}

182impl Verifier<state::Setup> {
183    /// Runs the verifier until the TLS connection is closed.
184    #[instrument(parent = &self.span, level = "info", skip_all, err)]
185    pub async fn run(self) -> Result<Verifier<state::Committed>, VerifierError> {
186        let state::Setup {
187            mux_ctrl,
188            mut mux_fut,
189            mpc_tls,
190            vm,
191            keys,
192        } = self.state;
193
194        info!("starting MPC-TLS");
195
196        let (mut ctx, tls_transcript) = mux_fut.poll_with(mpc_tls.run()).await?;
197
198        info!("finished MPC-TLS");
199
200        {
201            let mut vm = vm.try_lock().expect("VM should not be locked");
202
203            debug!("finalizing mpc");
204
205            mux_fut
206                .poll_with(vm.finalize(&mut ctx))
207                .await
208                .map_err(VerifierError::mpc)?;
209
210            debug!("mpc finalized");
211        }
212
213        // Pull out ZK VM.
214        let (_, mut vm) = Arc::into_inner(vm)
215            .expect("vm should have only 1 reference")
216            .into_inner()
217            .into_inner();
218
219        // Prepare for the prover to prove tag verification of the received
220        // records.
221        let tag_proof = verify_tags(
222            &mut vm,
223            (keys.server_write_key, keys.server_write_iv),
224            keys.server_write_mac_key,
225            *tls_transcript.version(),
226            tls_transcript.recv().to_vec(),
227        )
228        .map_err(VerifierError::zk)?;
229
230        mux_fut
231            .poll_with(vm.execute_all(&mut ctx).map_err(VerifierError::zk))
232            .await?;
233
234        // Verify the tags.
235        // After the verification, the entire TLS trancript becomes
236        // authenticated from the verifier's perspective.
237        tag_proof.verify().map_err(VerifierError::zk)?;
238
239        Ok(Verifier {
240            config: self.config,
241            span: self.span,
242            state: state::Committed {
243                mux_ctrl,
244                mux_fut,
245                ctx,
246                vm,
247                keys,
248                tls_transcript,
249            },
250        })
251    }
252}

254impl Verifier<state::Committed> {
255    /// Returns the TLS transcript.
256    pub fn tls_transcript(&self) -> &TlsTranscript {
257        &self.state.tls_transcript
258    }
259
260    /// Begins verification of statements from the prover.
261    #[instrument(parent = &self.span, level = "info", skip_all, err)]
262    pub async fn verify(self) -> Result<Verifier<state::Verify>, VerifierError> {
263        let state::Committed {
264            mux_ctrl,
265            mut mux_fut,
266            mut ctx,
267            vm,
268            keys,
269            tls_transcript,
270        } = self.state;
271
272        let request = mux_fut
273            .poll_with(ctx.io_mut().expect_next().map_err(VerifierError::from))
274            .await?;
275
276        Ok(Verifier {
277            config: self.config,
278            span: self.span,
279            state: state::Verify {
280                mux_ctrl,
281                mux_fut,
282                ctx,
283                vm,
284                keys,
285                tls_transcript,
286                request,
287            },
288        })
289    }
290
291    /// Closes the connection with the prover.
292    #[instrument(parent = &self.span, level = "info", skip_all, err)]
293    pub async fn close(self) -> Result<(), VerifierError> {
294        let state::Committed {
295            mux_ctrl, mux_fut, ..
296        } = self.state;
297
298        // Wait for the prover to correctly close the connection.
299        if !mux_fut.is_complete() {
300            mux_ctrl.close();
301            mux_fut.await?;
302        }
303
304        Ok(())
305    }
306}

308impl Verifier<state::Verify> {
309    /// Returns the proving request.
310    pub fn request(&self) -> &ProveRequest {
311        &self.state.request
312    }
313
314    /// Accepts the proving request.
315    pub async fn accept(
316        self,
317    ) -> Result<(VerifierOutput, Verifier<state::Committed>), VerifierError> {
318        let state::Verify {
319            mux_ctrl,
320            mut mux_fut,
321            mut ctx,
322            mut vm,
323            keys,
324            tls_transcript,
325            request,
326        } = self.state;
327
328        mux_fut.poll_with(ctx.io_mut().send(Response::ok())).await?;
329
330        let cert_verifier = if let Some(root_store) = self.config.root_store() {
331            ServerCertVerifier::new(root_store).map_err(VerifierError::config)?
332        } else {
333            ServerCertVerifier::mozilla()
334        };
335
336        let output = mux_fut
337            .poll_with(verify::verify(
338                &mut ctx,
339                &mut vm,
340                &keys,
341                &cert_verifier,
342                &tls_transcript,
343                request,
344            ))
345            .await?;
346
347        Ok((
348            output,
349            Verifier {
350                config: self.config,
351                span: self.span,
352                state: state::Committed {
353                    mux_ctrl,
354                    mux_fut,
355                    ctx,
356                    vm,
357                    keys,
358                    tls_transcript,
359                },
360            },
361        ))
362    }
363
364    /// Rejects the proving request.
365    pub async fn reject(
366        self,
367        msg: Option<&str>,
368    ) -> Result<Verifier<state::Committed>, VerifierError> {
369        let state::Verify {
370            mux_ctrl,
371            mut mux_fut,
372            mut ctx,
373            vm,
374            keys,
375            tls_transcript,
376            ..
377        } = self.state;
378
379        mux_fut
380            .poll_with(ctx.io_mut().send(Response::err(msg)))
381            .await?;
382
383        Ok(Verifier {
384            config: self.config,
385            span: self.span,
386            state: state::Committed {
387                mux_ctrl,
388                mux_fut,
389                ctx,
390                vm,
391                keys,
392                tls_transcript,
393            },
394        })
395    }
